14inline Eigen::MatrixXd convert_var_to_vma(Eigen::MatrixXd var_coef, int var_lag, int lag_max) {
15 int dim = var_coef.cols(); // m
16 if (lag_max < 1) {
17 STOP("'lag_max' must larger than 0");
18 }
19 int ma_rows = dim * (lag_max + 1);
20 int num_full_arows = ma_rows;
21 if (lag_max < var_lag) {
22 num_full_arows = dim * var_lag; // for VMA coefficient q < VAR(p)
23 }
24 Eigen::MatrixXd FullA = Eigen::MatrixXd::Zero(num_full_arows, dim); // same size with VMA coefficient matrix
25 FullA.block(0, 0, dim * var_lag, dim) = var_coef.block(0, 0, dim * var_lag, dim); // fill first mp row with VAR coefficient matrix
26 Eigen::MatrixXd Im = Eigen::MatrixXd::Identity(dim, dim); // identity matrix
27 Eigen::MatrixXd ma = Eigen::MatrixXd::Zero(ma_rows, dim); // VMA [W1^T, W2^T, ..., W(lag_max)^T]^T, ma_rows = m * lag_max
28 ma.block(0, 0, dim, dim) = Im; // W0 = Im
29 ma.block(dim, 0, dim, dim) = FullA.block(0, 0, dim, dim) * ma.block(0, 0, dim, dim); // W1^T = B1^T * W1^T
30 if (lag_max == 1) {
31 return ma;
32 }
33 for (int i = 2; i < (lag_max + 1); i++) { // from W2: m-th row
34 for (int k = 0; k < i; k++) {
35 ma.block(i * dim, 0, dim, dim) += FullA.block(k * dim, 0, dim, dim) * ma.block((i - k - 1) * dim, 0, dim, dim); // Wi = sum(W(i - k)^T * Bk^T)
36 }
37 }
38 return ma;
39}

41inline Eigen::MatrixXd convert_vma_ortho(Eigen::MatrixXd var_coef, Eigen::MatrixXd var_covmat, int var_lag, int lag_max) {
42 int dim = var_covmat.cols(); // num_rows = num_cols
43 if ((dim != var_covmat.rows()) && (dim != var_coef.cols())) {
44 STOP("Wrong covariance matrix format: `var_covmat`.");
45 }
46 if ((var_coef.rows() != var_lag * dim + 1) && (var_coef.rows() != var_lag * dim)) {
47 STOP("Wrong VAR coefficient format: `var_coef`.");
48 }
49 Eigen::MatrixXd ma = convert_var_to_vma(var_coef, var_lag, lag_max);
50 Eigen::MatrixXd res(ma.rows(), dim);
51 Eigen::LLT<Eigen::MatrixXd> lltOfcovmat(Eigen::Map<Eigen::MatrixXd>(var_covmat.data(), dim, dim)); // cholesky decomposition for Sigma
52 Eigen::MatrixXd chol_covmat = lltOfcovmat.matrixU();
53 for (int i = 0; i < lag_max + 1; i++) {
54 res.block(i * dim, 0, dim, dim) = chol_covmat * ma.block(i * dim, 0, dim, dim);

59inline Eigen::MatrixXd convert_vhar_to_vma(Eigen::MatrixXd vhar_coef, Eigen::MatrixXd HARtrans_mat, int lag_max, int month) {
60 int dim = vhar_coef.cols(); // dimension of time series
61 Eigen::MatrixXd coef_mat = HARtrans_mat.transpose() * vhar_coef; // bhat = tilde(T)^T * Phi
62 if (lag_max < 1) {
63 STOP("'lag_max' must larger than 0");
64 }
65 int ma_rows = dim * (lag_max + 1);
66 int num_full_arows = ma_rows;
67 if (lag_max < month) num_full_arows = month * dim; // for VMA coefficient q < VAR(p)
68 Eigen::MatrixXd FullA = Eigen::MatrixXd::Zero(num_full_arows, dim); // same size with VMA coefficient matrix
69 FullA.block(0, 0, month * dim, dim) = coef_mat.block(0, 0, month * dim, dim); // fill first mp row with VAR coefficient matrix
70 Eigen::MatrixXd Im = Eigen::MatrixXd::Identity(dim, dim); // identity matrix
71 Eigen::MatrixXd ma = Eigen::MatrixXd::Zero(ma_rows, dim); // VMA [W1^T, W2^T, ..., W(lag_max)^T]^T, ma_rows = m * lag_max
72 ma.block(0, 0, dim, dim) = Im; // W0 = Im
73 ma.block(dim, 0, dim, dim) = FullA.block(0, 0, dim, dim) * ma.block(0, 0, dim, dim); // W1^T = B1^T * W1^T
74 if (lag_max == 1) return ma;
75 for (int i = 2; i < (lag_max + 1); i++) { // from W2: m-th row
76 for (int k = 0; k < i; k++) {
77 ma.block(i * dim, 0, dim, dim) += FullA.block(k * dim, 0, dim, dim) * ma.block((i - k - 1) * dim, 0, dim, dim); // Wi = sum(W(i - k)^T * Bk^T)
78 }
79 }
80 return ma;
81}

101inline Eigen::MatrixXd compute_vma_fevd(Eigen::MatrixXd vma_coef, Eigen::MatrixXd cov_mat, bool normalize) {
102 int dim = cov_mat.cols();
103 // Eigen::MatrixXd vma_mat = VARcoeftoVMA(var_coef, var_lag, step);
104 int step = vma_coef.rows() / dim; // h-step
105 Eigen::MatrixXd innov_account = Eigen::MatrixXd::Zero(dim, dim);
106 Eigen::MatrixXd ma_prod(dim, dim);
107 Eigen::MatrixXd numer = Eigen::MatrixXd::Zero(dim, dim);
108 Eigen::MatrixXd denom = Eigen::MatrixXd::Zero(dim, dim);
109 Eigen::MatrixXd res = Eigen::MatrixXd::Zero(dim * step, dim);
110 Eigen::MatrixXd cov_diag = Eigen::MatrixXd::Zero(dim, dim);
111 cov_diag.diagonal() = 1 / cov_mat.diagonal().cwiseSqrt().array(); // sigma_jj
112 for (int i = 0; i < step; i++) {
113 ma_prod = vma_coef.block(i * dim, 0, dim, dim).transpose() * cov_mat; // A * Sigma
114 innov_account += ma_prod * vma_coef.block(i * dim, 0, dim, dim); // A * Sigma * A^T
115 numer.array() += (ma_prod * cov_diag).array().square(); // sum(A * Sigma)_ij / sigma_jj^2
116 denom.diagonal() = 1 / innov_account.diagonal().array(); // sigma_jj^(-1) / sum(A * Sigma * A^T)_jj
117 res.block(i * dim, 0, dim, dim) = denom * numer; // sigma_jj^(-1) sum(A * Sigma)_ij / sum(A * Sigma * A^T)_jj
118 }
119 if (normalize) {
120 res.array().colwise() /= res.rowwise().sum().array();
121 }
122 return res;
123}
