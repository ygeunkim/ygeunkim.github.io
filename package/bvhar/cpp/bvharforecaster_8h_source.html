598  void runGibbs(int window, int chain) {
599  std::string log_name = fmt::format("Chain {} / Window {}", chain + 1, window + 1);
600  auto logger = SPDLOG_SINK_MT(log_name);
601  logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
602  int logging_freq = num_iter / 20; // 5 percent
603  if (logging_freq == 0) {
604  logging_freq = 1;
605  }
606  bvharinterrupt();
607  for (int i = 0; i < num_burn; ++i) {
608  model[window][chain]->doWarmUp();
609  if (display_progress && (i + 1) % logging_freq == 0) {
610  logger->info("{} / {} (Warmup)", i + 1, num_iter);
611  }
612  }
613  logger->flush();
614  for (int i = num_burn; i < num_iter; ++i) {
615  if (bvharinterrupt::is_interrupted()) {
616  RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
617  logger->warn("User interrupt in {} / {}", i + 1, num_iter);
618  break;
619  }
620  model[window][chain]->doPosteriorDraws();
621  if (display_progress && (i + 1) % logging_freq == 0) {
622  logger->info("{} / {} (Sampling)", i + 1, num_iter);
623  }
624  }
625  RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
626  updateForecaster(reg_record, window, chain);
627  model[window][chain].reset();
628  logger->flush();
629  spdlog::drop(log_name);
630  }
631  void forecastWindow(int window, int chain) {
632  if (window != 0) {
633  runGibbs(window, chain);
634  }
635  Eigen::VectorXd valid_vec = y_test.row(step);
636  out_forecast[window][chain] = forecaster[window][chain]->forecastDensity(valid_vec).bottomRows(1);
637  lpl_record(window, chain) = forecaster[window][chain]->returnLpl();
638  forecaster[window][chain].reset(); // free the memory by making nullpt
639  }
