9 // Numerically Stable Log Marginal Likelihood Excluding Constant Term
10 //
11 // This function computes log of ML stable,
12 // excluding the constant term.
13 //
14 // @param dim Dimension of the time series
15 // @param num_design The number of the data matrix, \eqn{n = T - p}
16 // @param prior_prec Prior precision of Matrix Normal distribution
17 // @param prior_scale Prior scale of Inverse-Wishart distribution
18 // @param mn_prec Posterior precision of Matrix Normal distribution
19 // @param iw_scale Posterior scale of Inverse-Wishart distribution
20 // @param posterior_shape Posterior shape of Inverse-Wishart distribution
21 inline double compute_logml(int dim, int num_design, Eigen::MatrixXd prior_prec, Eigen::MatrixXd prior_scale,
22  Eigen::MatrixXd mn_prec, Eigen::MatrixXd iw_scale, int posterior_shape) {
23  Eigen::LLT<Eigen::MatrixXd> lltOfmn(prior_prec.inverse());
24  Eigen::MatrixXd chol_mn = lltOfmn.matrixL();
25  Eigen::MatrixXd stable_mat_a = chol_mn.transpose() * (mn_prec - prior_prec) * chol_mn;
26  Eigen::LLT<Eigen::MatrixXd> lltOfiw(prior_scale.inverse());
27  Eigen::MatrixXd chol_iw = lltOfiw.matrixL();
28  Eigen::MatrixXd stable_mat_b = chol_iw.transpose() * (iw_scale - prior_scale) * chol_iw;
29  // eigenvalues
30  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es_a(stable_mat_a);
31  Eigen::VectorXd a_eigen = es_a.eigenvalues();
32  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es_b(stable_mat_b);
33  Eigen::VectorXd b_eigen = es_b.eigenvalues();
34  // sum of log(1 + eigenvalues)
35  double a_term = a_eigen.array().log1p().sum();
36  double b_term = b_eigen.array().log1p().sum();
37  // result
38  return - num_design / 2.0 * log(prior_scale.determinant()) - dim / 2.0 * a_term - posterior_shape / 2.0 * b_term;
39 }

51 // @param mn_prec Posterior precision of Matrix Normal distribution
52 // @param iw_scale Posterior scale of Inverse-Wishart distribution
53 // @param posterior_shape Posterior shape of Inverse-Wishart distribution
54 // @param gamma_shape Shape of hyperprior Gamma distribution
55 // @param gamma_rate Rate of hyperprior Gamma distribution
56 // @param invgam_shape Shape of hyperprior Inverse gamma distribution
57 // @param invgam_scl Scale of hyperprior Inverse gamma distribution
58 inline double jointdens_hyperparam(double cand_gamma, Eigen::VectorXd cand_invgam, int dim, int num_design,

61  int posterior_shape, double gamma_shp, double gamma_rate, double invgam_shp, double invgam_scl) {

420 // Generating log-volatilities in MCMC
421 //
422 // In MCMC, this function samples log-volatilities \eqn{h_{it}} vector using auxiliary mixture sampling
423 //
424 // @param sv_vec log-volatilities vector
425 // @param init_sv Initial log-volatility
426 // @param sv_sig Variance of log-volatilities
427 // @param latent_vec Auxiliary residual vector
428 inline void varsv_ht(Eigen::Ref<Eigen::VectorXd> sv_vec, double init_sv,
429  double sv_sig, Eigen::Ref<Eigen::VectorXd> latent_vec, boost::random::mt19937& rng) {
430  int num_design = sv_vec.size(); // h_i1, ..., h_in for i = 1, .., k
431  // 7-component normal mixutre
432  Eigen::VectorXd pj(7); // p_t
433  pj << 0.0073, 0.10556, 0.00002, 0.04395, 0.34001, 0.24566, 0.2575;
434  Eigen::VectorXd muj(7); // mu_t
435  muj << -10.12999, -3.97281, -8.56686, 2.77786, 0.61942, 1.79518, -1.08819;
436  muj.array() -= 1.2704;
437  Eigen::VectorXd sigj(7); // sig_t^2
438  sigj << 5.79596, 2.61369, 5.17950, 0.16735, 0.64009, 0.34023, 1.26261;
439  Eigen::VectorXd sdj = sigj.cwiseSqrt();
440  Eigen::VectorXi binom_latent(num_design);
441  Eigen::VectorXd ds(num_design); // (mu_st - 1.2704)
442  Eigen::MatrixXd inv_sig_s = Eigen::MatrixXd::Zero(num_design, num_design); // diag(1 / sig_st^2)
443  Eigen::VectorXd inv_method(num_design); // inverse transform method
444  Eigen::MatrixXd mixture_pdf(num_design, 7);
445  Eigen::MatrixXd mixture_cumsum = Eigen::MatrixXd::Zero(num_design, 7);
446  for (int i = 0; i < num_design; i++) {
447  inv_method[i] = unif_rand(0, 1, rng);
448  }
449  for (int i = 0; i < 7; i++) {
450  mixture_pdf.col(i) = (-((latent_vec.array() - sv_vec.array() - muj[i]).array() / sdj[i]).array().square() / 2).exp() * pj[i] / (sdj[i] * sqrt(2 * M_PI));
451  }
452  mixture_pdf.array().colwise() /= mixture_pdf.rowwise().sum().array();
453  for (int i = 0; i < 7; i++) {
454  mixture_cumsum.block(0, i, num_design, 7 - i) += mixture_pdf.col(i).rowwise().replicate(7 - i);
455  }
456  binom_latent.array() = 7 - (inv_method.rowwise().replicate(7).array() < mixture_cumsum.array()).cast<int>().rowwise().sum().array(); // 0 to 6 for indexing
457  Eigen::MatrixXd diff_mat = Eigen::MatrixXd::Identity(num_design, num_design);
458  for (int i = 0; i < num_design - 1; i++) {
459  ds[i] = muj[binom_latent[i]];
460  inv_sig_s(i, i) = 1 / sigj[binom_latent[i]];
461  diff_mat(i + 1, i) = -1;
462  }
463  ds[num_design - 1] = muj[binom_latent[num_design - 1]];
464  inv_sig_s(num_design - 1, num_design - 1) = 1 / sigj[binom_latent[num_design - 1]];
465  Eigen::MatrixXd HtH = diff_mat.transpose() * diff_mat;
466  Eigen::VectorXd res(num_design);
467  for (int i = 0; i < num_design; i++) {
468  res[i] = normal_rand(rng);
469  }
470  Eigen::MatrixXd post_sig = HtH / sv_sig + inv_sig_s;
471  Eigen::LLT<Eigen::MatrixXd> lltOfscale(post_sig);
472  Eigen::VectorXd post_mean = lltOfscale.solve(
473  HtH * init_sv * Eigen::VectorXd::Ones(num_design) / sv_sig + inv_sig_s * (latent_vec - ds)
474  );
475  sv_vec = post_mean + lltOfscale.matrixU().solve(res);
476 }
