43 void computeSpillover() {
44 for (int i = 0; i < num_sim; ++i) {
45 reg_record->updateDiag(i, time_id, sv_update);
46 sqrt_sig = build_inv_lower(
47 dim,
48 reg_record->contem_coef_record.row(i)
49 ).triangularView<Eigen::UnitLower>().solve(sv_update.asDiagonal().toDenseMatrix());
50 cov = sqrt_sig * sqrt_sig.transpose();
51 coef_mat = unvectorize(reg_record->coef_record.row(i).transpose(), dim);
52 computeVma();
53 fevd.middleCols(i * dim, dim) = compute_vma_fevd(vma_mat, cov, true);
54 spillover.middleCols(i * dim, dim) = compute_sp_index(fevd.middleCols(i * dim, dim));
55 to_spillover.segment(i * dim, dim) = compute_to(spillover.middleCols(i * dim, dim));
56 from_spillover.segment(i * dim, dim) = compute_from(spillover.middleCols(i * dim, dim));
57 tot_spillover[i] = compute_tot(spillover.middleCols(i * dim, dim));
58 net_spillover.middleCols(i * dim, dim) = compute_net(spillover.middleCols(i * dim, dim));
59 }

412 void getSpillover(int window, int chain) {

420 void fit() {
421 if (num_chains == 1) {
422 #ifdef _OPENMP
423 #pragma omp parallel for num_threads(nthreads)
424 #endif
425 for (int window = 0; window < num_horizon; ++window) {
426 runGibbs(window, 0);
427 getSpillover(window, 0);
428 }
429 } else {
430 #ifdef _OPENMP
431 #pragma omp parallel for collapse(2) schedule(static, num_chains) num_threads(nthreads)
432 #endif
433 for (int window = 0; window < num_horizon; ++window) {
434 for (int chain = 0; chain < num_chains; ++chain) {
