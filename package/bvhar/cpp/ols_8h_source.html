24struct StructuralFit : public OlsFit {
25 int _lag_max;
26 int dim;
27 int ma_rows;
28 Eigen::MatrixXd _vma; // VMA [W1^T, W2^T, ..., W(lag_max)^T]^T, ma_rows = m * lag_max
29 Eigen::MatrixXd _cov;
30
31 StructuralFit(const Eigen::MatrixXd& coef_mat, int ord, int lag_max, const Eigen::MatrixXd& cov_mat)
32 : OlsFit(coef_mat, ord), _lag_max(lag_max),
33 dim(coef_mat.cols()), ma_rows(dim * (_lag_max + 1)),
34 _vma(Eigen::MatrixXd::Zero(ma_rows, dim)), _cov(cov_mat) {
35 int num_full_rows = _lag_max < _ord ? dim * _ord : ma_rows;
36 Eigen::MatrixXd full_coef = Eigen::MatrixXd::Zero(num_full_rows, dim); // same size with VMA coefficient matrix
37 full_coef.topRows(dim * _ord) = _coef.topRows(dim * _ord); // fill first mp row with VAR coefficient matrix
38 _vma.topRows(dim) = Eigen::MatrixXd::Identity(dim, dim); // W0 = I_k
39 for (int i = 1; i < (_lag_max + 1); ++i) {
40 for (int j = 0; j < i; ++j) {
41 _vma.middleRows(i * dim, dim) += full_coef.middleRows(j * dim, dim) * _vma.middleRows((i - j - 1) * dim, dim); // Wi = sum(W(i - k)^T * Bk^T)
42 }
43 }
44 }

47class MultiOls {
48public:
49 MultiOls(const Eigen::MatrixXd& x, const Eigen::MatrixXd& y)
50 : design(x), response(y),
51 dim(response.cols()), num_design(response.rows()), dim_design(design.cols()) {
52 coef = Eigen::MatrixXd::Zero(dim_design, dim);
53 yhat = Eigen::MatrixXd::Zero(num_design, dim);
54 resid = Eigen::MatrixXd::Zero(num_design, dim);
55 cov = Eigen::MatrixXd::Zero(dim, dim);
56 }
57 virtual ~MultiOls() = default;
58 virtual void estimateCoef() {
59 coef = (design.transpose() * design).inverse() * design.transpose() * response; // return coef -> use in OlsVar
60 }
61 virtual void fitObs() {
62 yhat = design * coef;
63 resid = response - yhat;
64 }
65 void estimateCov() {
66 cov = resid.transpose() * resid / (num_design - dim_design);
67 }
68 LIST returnOlsRes() {

70 fitObs();
71 estimateCov();
72 return CREATE_LIST(
73 NAMED("coefficients") = coef,
74 NAMED("fitted.values") = yhat,
75 NAMED("residuals") = resid,
76 NAMED("covmat") = cov,
77 NAMED("df") = dim_design,
78 NAMED("m") = dim,
79 NAMED("obs") = num_design,
80 NAMED("y0") = response
81 );
82 }

107 int dim; // k
108 int num_design; // n
