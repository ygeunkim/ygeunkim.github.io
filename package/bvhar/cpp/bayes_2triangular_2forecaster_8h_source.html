755 void runGibbs(int window, int chain) {
756 std::string log_name = fmt::format("Chain {} / Window {}", chain + 1, window + 1);
757 auto logger = spdlog::get(log_name);
758 if (logger == nullptr) {
759 logger = SPDLOG_SINK_MT(log_name);
760 }
761 logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
762 int logging_freq = num_iter / 20; // 5 percent
763 if (logging_freq == 0) {
764 logging_freq = 1;
765 }
766 bvharinterrupt();
767 for (int i = 0; i < num_burn; ++i) {
768 model[window][chain]->doWarmUp();
769 if (display_progress && (i + 1) % logging_freq == 0) {
770 logger->info("{} / {} (Warmup)", i + 1, num_iter);
771 }
772 }
773 logger->flush();
774 for (int i = num_burn; i < num_iter; ++i) {
775 if (bvharinterrupt::is_interrupted()) {
776 RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
777 logger->warn("User interrupt in {} / {}", i + 1, num_iter);
778 break;
779 }
780 model[window][chain]->doPosteriorDraws();
781 if (display_progress && (i + 1) % logging_freq == 0) {
782 logger->info("{} / {} (Sampling)", i + 1, num_iter);
783 }
784 }
785 RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
786 updateForecaster(reg_record, window, chain);
787 model[window][chain].reset();
788 logger->flush();
789 spdlog::drop(log_name);
790 }
