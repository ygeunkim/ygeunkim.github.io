152public:
153 Minnesota(const Eigen::MatrixXd& x, const Eigen::MatrixXd& y, const Eigen::MatrixXd& x_dummy, const Eigen::MatrixXd& y_dummy)
154 : design(x), response(y), dummy_design(x_dummy), dummy_response(y_dummy),
155 dim(response.cols()), num_design(response.rows()), dim_design(design.cols()),
156 num_dummy(dummy_design.rows()), num_augment(num_design + num_dummy) {
157 prior_prec = dummy_design.transpose() * dummy_design;
158 prior_mean = prior_prec.inverse() * dummy_design.transpose() * dummy_response;
159 prior_scale = (dummy_response - dummy_design * prior_mean).transpose() * (dummy_response - dummy_design * prior_mean);
160 prior_shape = num_dummy - dim_design + 2;
161 ystar = Eigen::MatrixXd::Zero(num_augment, dim);
162 ystar << y,
163 y_dummy;
164 xstar = Eigen::MatrixXd::Zero(num_augment, dim_design);
165 xstar << x,
166 x_dummy;

168 prec = Eigen::MatrixXd::Zero(dim, dim);
169 yhat = Eigen::MatrixXd::Zero(num_design, dim);

171 yhat_star = Eigen::MatrixXd::Zero(num_augment, dim);
172 scale = Eigen::MatrixXd::Zero(dim, dim);
173 }

176 prec = xstar.transpose() * xstar;

180 yhat = design * coef;
181 resid = response - yhat;
182 }
183 void estimateCov() { // Posterior IW scale
184 yhat_star = xstar * coef;
185 scale = (ystar - yhat_star).transpose() * (ystar - yhat_star);
186 }
