193  void updateCoef() {
194  for (int j = 0; j < dim; ++j) {
195  coef_mat.col(j).setZero(); // j-th column of A = 0
196  Eigen::MatrixXd chol_lower_j = chol_lower.bottomRows(dim - j); // L_(j:k) = a_jt to a_kt for t = 1, ..., j - 1
197  Eigen::MatrixXd sqrt_sv_j = sqrt_sv.rightCols(dim - j); // use h_jt to h_kt for t = 1, .. n => (k - j + 1) x k
198  Eigen::MatrixXd design_coef = kronecker_eigen(chol_lower_j.col(j), x).array().colwise() / sqrt_sv_j.reshaped().array(); // L_(j:k, j) otimes X0 scaled by D_(1:n, j:k): n(k - j + 1) x kp
199  Eigen::VectorXd prior_mean_j(dim_design);
200  Eigen::VectorXd prior_prec_j(dim_design);
201  Eigen::VectorXd penalty_j = Eigen::VectorXd::Zero(dim_design);
202  if (include_mean) {
203  prior_mean_j << prior_alpha_mean.segment(j * nrow_coef, nrow_coef), prior_alpha_mean.tail(dim)[j];
204  prior_prec_j << prior_alpha_prec.segment(j * nrow_coef, nrow_coef), prior_alpha_prec.tail(dim)[j];
205  // penalty_j << alpha_penalty.segment(j * nrow_coef, nrow_coef), alpha_penalty.tail(dim)[j];
206  penalty_j.head(nrow_coef) = alpha_penalty.segment(j * nrow_coef, nrow_coef);
207  draw_coef(
208  coef_mat.col(j), design_coef,
209  (((y - x * coef_mat) * chol_lower_j.transpose()).array() / sqrt_sv_j.array()).reshaped(), // Hadamard product between: (Y - X0 A(-j))L_(j:k)^T and D_(1:n, j:k)
210  prior_mean_j, prior_prec_j, rng
211  );
212  coef_vec.head(num_alpha) = coef_mat.topRows(nrow_coef).reshaped();
213  coef_vec.tail(dim) = coef_mat.bottomRows(1).transpose();
214  } else {
215  prior_mean_j = prior_alpha_mean.segment(dim_design * j, dim_design);
216  prior_prec_j = prior_alpha_prec.segment(dim_design * j, dim_design);
217  penalty_j = alpha_penalty.segment(dim_design * j, dim_design);
218  draw_coef(
219  coef_mat.col(j),
220  design_coef,
221  (((y - x * coef_mat) * chol_lower_j.transpose()).array() / sqrt_sv_j.array()).reshaped(),
222  prior_mean_j, prior_prec_j, rng
223  );
224  coef_vec = coef_mat.reshaped();
225  }
226  draw_mn_savs(sparse_coef.col(j), coef_mat.col(j), x, penalty_j);
227  }
228  }
229  void updateImpact() {
230  for (int j = 1; j < dim; ++j) {
231  response_contem = latent_innov.col(j).array() / sqrt_sv.col(j).array(); // n-dim
232  Eigen::MatrixXd design_contem = latent_innov.leftCols(j).array().colwise() / sqrt_sv.col(j).reshaped().array(); // n x (j - 1)
233  contem_id = j * (j - 1) / 2;
234  draw_coef(
235  contem_coef.segment(contem_id, j),
236  design_contem, response_contem,

238  prior_chol_prec.segment(contem_id, j),
239  rng
240  );

242  }
243  }
