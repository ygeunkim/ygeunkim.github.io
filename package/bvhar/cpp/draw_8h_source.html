10// Numerically Stable Log Marginal Likelihood Excluding Constant Term
11//
12// This function computes log of ML stable,
13// excluding the constant term.
14//
15// @param dim Dimension of the time series
16// @param num_design The number of the data matrix, \eqn{n = T - p}
17// @param prior_prec Prior precision of Matrix Normal distribution
18// @param prior_scale Prior scale of Inverse-Wishart distribution
19// @param mn_prec Posterior precision of Matrix Normal distribution
20// @param iw_scale Posterior scale of Inverse-Wishart distribution
21// @param posterior_shape Posterior shape of Inverse-Wishart distribution
22inline double compute_logml(int dim, int num_design, Eigen::MatrixXd prior_prec, Eigen::MatrixXd prior_scale,
23 Eigen::MatrixXd mn_prec, Eigen::MatrixXd iw_scale, int posterior_shape) {
24 Eigen::LLT<Eigen::MatrixXd> lltOfmn(prior_prec.inverse());
25 Eigen::MatrixXd chol_mn = lltOfmn.matrixL();
26 Eigen::MatrixXd stable_mat_a = chol_mn.transpose() * (mn_prec - prior_prec) * chol_mn;
27 Eigen::LLT<Eigen::MatrixXd> lltOfiw(prior_scale.inverse());
28 Eigen::MatrixXd chol_iw = lltOfiw.matrixL();
29 Eigen::MatrixXd stable_mat_b = chol_iw.transpose() * (iw_scale - prior_scale) * chol_iw;
30 // eigenvalues
31 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es_a(stable_mat_a);
32 Eigen::VectorXd a_eigen = es_a.eigenvalues();
33 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es_b(stable_mat_b);
34 Eigen::VectorXd b_eigen = es_b.eigenvalues();
35 // sum of log(1 + eigenvalues)
36 double a_term = a_eigen.array().log1p().sum();
37 double b_term = b_eigen.array().log1p().sum();
38 // result
39 return - num_design / 2.0 * log(prior_scale.determinant()) - dim / 2.0 * a_term - posterior_shape / 2.0 * b_term;
40}

42// Log of Joint Posterior Density of Hyperparameters
43//
44// This function computes the log of joint posterior density of hyperparameters.
45//
46// @param cand_gamma Candidate value of hyperparameters following Gamma distribution
47// @param cand_invgam Candidate value of hyperparameters following Inverse Gamma distribution
48// @param dim Dimension of the time series
49// @param num_design The number of the data matrix, \eqn{n = T - p}
50// @param prior_prec Prior precision of Matrix Normal distribution
51// @param prior_scale Prior scale of Inverse-Wishart distribution
52// @param mn_prec Posterior precision of Matrix Normal distribution
53// @param iw_scale Posterior scale of Inverse-Wishart distribution
54// @param posterior_shape Posterior shape of Inverse-Wishart distribution
55// @param gamma_shape Shape of hyperprior Gamma distribution
56// @param gamma_rate Rate of hyperprior Gamma distribution
57// @param invgam_shape Shape of hyperprior Inverse gamma distribution
58// @param invgam_scl Scale of hyperprior Inverse gamma distribution
59inline double jointdens_hyperparam(double cand_gamma, Eigen::VectorXd cand_invgam, int dim, int num_design,
60 Eigen::MatrixXd prior_prec, Eigen::MatrixXd prior_scale, int prior_shape,
61 Eigen::MatrixXd mn_prec, Eigen::MatrixXd iw_scale,
62 int posterior_shape, double gamma_shp, double gamma_rate, double invgam_shp, double invgam_scl) {
63 double res = compute_logml(dim, num_design, prior_prec, prior_scale, mn_prec, iw_scale, posterior_shape);
64 res += -dim * num_design / 2.0 * log(M_PI) +
65 lmgammafn((prior_shape + num_design) / 2.0, dim) -
66 lmgammafn(prior_shape / 2.0, dim); // constant term
67 res += gamma_dens(cand_gamma, gamma_shp, 1 / gamma_rate, true); // gamma distribution
68 for (int i = 0; i < cand_invgam.size(); i++) {
69 res += invgamma_dens(cand_invgam[i], invgam_shp, invgam_scl, true); // inverse gamma distribution
70 }
71 return res;
72}

139// @param sse_mat The result of \eqn{Z_0^T Z_0 = (Y_0 - X_0 \hat{A})^T (Y_0 - X_0 \hat{A})}

149 inv_DRD.diagonal() = 1 / DRD.array().square();

262//

264//

432// Generating log-volatilities in MCMC
433//
434// In MCMC, this function samples log-volatilities \eqn{h_{it}} vector using auxiliary mixture sampling
435//
436// @param sv_vec log-volatilities vector
437// @param init_sv Initial log-volatility
438// @param sv_sig Variance of log-volatilities
439// @param latent_vec Auxiliary residual vector
440inline void varsv_ht(Eigen::Ref<Eigen::VectorXd> sv_vec, double init_sv,
441 double sv_sig, Eigen::Ref<Eigen::VectorXd> latent_vec, boost::random::mt19937& rng) {
442 int num_design = sv_vec.size(); // h_i1, ..., h_in for i = 1, .., k
443 // 7-component normal mixutre
444 Eigen::VectorXd pj(7); // p_t
445 pj << 0.0073, 0.10556, 0.00002, 0.04395, 0.34001, 0.24566, 0.2575;
446 Eigen::VectorXd muj(7); // mu_t
447 muj << -10.12999, -3.97281, -8.56686, 2.77786, 0.61942, 1.79518, -1.08819;
448 muj.array() -= 1.2704;
449 Eigen::VectorXd sigj(7); // sig_t^2
450 sigj << 5.79596, 2.61369, 5.17950, 0.16735, 0.64009, 0.34023, 1.26261;
451 Eigen::VectorXd sdj = sigj.cwiseSqrt();
452 Eigen::VectorXi binom_latent(num_design);
453 Eigen::VectorXd ds(num_design); // (mu_st - 1.2704)
454 Eigen::MatrixXd inv_sig_s = Eigen::MatrixXd::Zero(num_design, num_design); // diag(1 / sig_st^2)
455 Eigen::VectorXd inv_method(num_design); // inverse transform method
456 Eigen::MatrixXd mixture_pdf(num_design, 7);
457 Eigen::MatrixXd mixture_cumsum = Eigen::MatrixXd::Zero(num_design, 7);
458 for (int i = 0; i < num_design; i++) {
459 inv_method[i] = unif_rand(0, 1, rng);
460 }
461 for (int i = 0; i < 7; i++) {
462 mixture_pdf.col(i) = (-((latent_vec.array() - sv_vec.array() - muj[i]).array() / sdj[i]).array().square() / 2).exp() * pj[i] / (sdj[i] * sqrt(2 * M_PI));
463 }
464 mixture_pdf.array().colwise() /= mixture_pdf.rowwise().sum().array();
465 for (int i = 0; i < 7; i++) {
466 mixture_cumsum.block(0, i, num_design, 7 - i) += mixture_pdf.col(i).rowwise().replicate(7 - i);
467 }
468 binom_latent.array() = 7 - (inv_method.rowwise().replicate(7).array() < mixture_cumsum.array()).cast<int>().rowwise().sum().array(); // 0 to 6 for indexing
469 Eigen::MatrixXd diff_mat = Eigen::MatrixXd::Identity(num_design, num_design);
470 for (int i = 0; i < num_design - 1; i++) {
471 ds[i] = muj[binom_latent[i]];
472 inv_sig_s(i, i) = 1 / sigj[binom_latent[i]];
473 diff_mat(i + 1, i) = -1;
474 }
475 ds[num_design - 1] = muj[binom_latent[num_design - 1]];
476 inv_sig_s(num_design - 1, num_design - 1) = 1 / sigj[binom_latent[num_design - 1]];
477 Eigen::MatrixXd HtH = diff_mat.transpose() * diff_mat;
478 Eigen::VectorXd res(num_design);
479 for (int i = 0; i < num_design; i++) {
480 res[i] = normal_rand(rng);
481 }
482 Eigen::MatrixXd post_sig = HtH / sv_sig + inv_sig_s;
483 Eigen::LLT<Eigen::MatrixXd> lltOfscale(post_sig);
484 Eigen::VectorXd post_mean = lltOfscale.solve(
485 HtH * init_sv * Eigen::VectorXd::Ones(num_design) / sv_sig + inv_sig_s * (latent_vec - ds)
486 );
487 sv_vec = post_mean + lltOfscale.matrixU().solve(res);
488}
