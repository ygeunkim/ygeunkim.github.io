32 // Compute mode of quasi-density of GIG
33 // @param lambda Index of modified Bessel function of third kind.
34 // @param beta Square of the multiplication of the other two parameters.
35 inline double dgig_mode(double lambda, double beta) {
36  if (lambda < 1) {
37  return beta / (sqrt((1 - lambda) * (1 - lambda) + beta * beta) + 1 - lambda);
38  }
39  return (sqrt((lambda - 1) * (lambda - 1) + beta * beta) - 1 + lambda) / beta;
40 }

158 // AR-Mehod for non-concave part
159 //
160 // Algorithm 1 of Hormann and Leydold
161 // 0 <= lambda < 1 and 0 < beta <= 2 / 3 * sqrt(1 - lambda)
162 //
163 // @param num_sim Number to generate process
164 // @param lambda Index of modified Bessel function of third kind.
165 // @param beta Square of the multiplication of the other two parameters.
166 inline void rgig_nonconcave(Eigen::VectorXd& res, int num_sim, double lambda, double beta) {
167  double mode = dgig_mode(lambda, beta); // argmax of g(x)
168  double x0, xstar, k1, k2, k3, A1, A2, A3;
169  x0 = beta / (1 - lambda); // subdomain (0, x0)
170  xstar = std::max(x0, 2 / beta);
171  k1 = exp(dgig_quasi(mode, lambda, beta));
172  A1 = k1 * x0;
173  if (x0 < 2 / beta) { // subdomain (x0, 2 / beta)
174  k2 = exp(-beta);
175  if (lambda == 0) {
176  A2 = k2 * log(2 / (beta * beta));
177  } else {
178  A2 = k2 * (pow(2 / beta, lambda) - pow(x0, lambda)) / lambda;
179  }
180  } else {
181  k2 = 0;
182  A2 = 0;
183  }
184  k3 = pow(xstar, lambda - 1); // subdomain (xstar, inf)
185  A3 = 2 * k3 * exp(-xstar * beta / 2) / beta;
186  double A = A1 + A2 + A3;
187  bool rejected;
188  double draw_unif, draw_prop, cand, ar_const;
189  for (int i = 0; i < num_sim; i++) {
190  rejected = true;
191  while (rejected) {
192  draw_prop = unif_rand(0, A);
193  if (draw_prop <= A1) { // subdomain (0, x0)
194  cand = x0 * draw_prop / A1;
195  ar_const = log(k1);
196  } else if (draw_prop <= A1 + A2) { // subdomain (x0, 2 / beta)
197  draw_prop -= A1;
198  if (lambda == 0) {
199  cand = beta * exp(draw_prop * exp(beta));
200  } else {
201  cand = pow(pow(x0, lambda) + draw_prop * lambda / k2, 1 / lambda);
202  }
203  ar_const = log(k2) + (lambda - 1) * log(cand);
204  } else { // subdomain (xstar, inf)
205  draw_prop -= (A1 + A2);
206  cand = -2 * log(exp(-xstar * beta / 2) - draw_prop * beta / (2 * k3)) / beta;
207  ar_const = log(k3) - cand * beta / 2;
208  }
209  draw_unif = unif_rand(0, 1);
210  rejected = log(draw_unif) + ar_const > dgig_quasi(cand, lambda, beta);
211  }
212  res[i] = cand;
213  }
214 }

216 // Ratio-of-Uniforms without Mode Shift
217 //
218 // Algorithm 2 of Hormann and Leydold
219 // 0 <= lambda <= 1 and min(1 / 2, 2 / 3 * sqrt(1 - lambda)) <= beta <= 1
220 //
221 // @param num_sim Number to generate process
222 // @param lambda Index of modified Bessel function of third kind.
223 // @param beta Square of the multiplication of the other two parameters.
224 inline void rgig_without_mode(Eigen::VectorXd& res, int num_sim, double lambda, double beta) {
225  double mode = dgig_mode(lambda, beta); // argmax of g(x)
226  double mode_x = (1 + lambda + sqrt((1 + lambda) * (1 + lambda) + beta * beta)) / beta; // argmax of x g(x)
227  double bound_y = dgig_quasi(mode, lambda, beta) / 2; // To normalize g
228  double bound_x = exp(log(mode_x) + dgig_quasi(mode_x, lambda, beta) / 2 - bound_y);
229  bool rejected;
230  double draw_x, draw_y, cand; // bounded rectangle
231  for (int i = 0; i < num_sim; i++) {
232  rejected = true;
233  while (rejected) {
234  draw_x = unif_rand(0, bound_x);
235  draw_y = unif_rand(0, 1);
236  cand = draw_x / draw_y;
237  rejected = log(draw_y) > dgig_quasi(cand, lambda, beta) / 2 - bound_y; // Check if U <= g(y) / unif(y)
238  }
239  res[i] = cand;
240  }
241 }

243 // Ratio-of-Uniforms with Mode Shift
244 //
245 // Algorithm 2 of Hormann and Leydold
246 // lambda > 1 and beta > 1
247 //
248 // @param num_sim Number to generate process
249 // @param lambda Index of modified Bessel function of third kind.
250 // @param beta Square of the multiplication of the other two parameters.
251 inline void rgig_with_mode(Eigen::VectorXd& res, int num_sim, double lambda, double beta) {
252  double mode = dgig_mode(lambda, beta); // argmax of g(x)
253  double quad_coef = -2 * (lambda + 1) / beta - mode;
254  double lin_coef = 2 * mode * (lambda - 1) / beta - 1;
255  double p = lin_coef - quad_coef * quad_coef / 3;
256  double q = 2 * quad_coef * quad_coef * quad_coef / 27 - quad_coef * lin_coef / 3 + mode;
257  double phi = acos(-q * sqrt(-27 / (p * p * p)) / 2);
258  double arg_x_neg = sqrt(-p * 4 / 3) * cos(phi / 3 + M_PI * 4 / 3) - quad_coef / 3;
259  double arg_x_pos = sqrt(-p * 4 / 3) * cos(phi / 3) - quad_coef / 3;
260  double bound_y = dgig_quasi(mode, lambda, beta) / 2; // use as normalize factor
261  double bound_x_neg = (arg_x_neg - mode) * exp(dgig_quasi(arg_x_neg, lambda, beta) / 2 - bound_y);
262  double bound_x_pos = (arg_x_pos - mode) * exp(dgig_quasi(arg_x_pos, lambda, beta) / 2 - bound_y);
263  bool rejected;
264  double draw_x, draw_y, cand; // bounded rectangle
265  for (int i = 0; i < num_sim; i++) {
266  rejected = true;
267  while (rejected) {
268  draw_x = unif_rand(bound_x_neg, bound_x_pos);
269  draw_y = unif_rand(0, 1); // U(0, 1) since g has been normalized
270  cand = draw_x / draw_y + mode;
271  if (cand > 0) {
272  rejected = log(draw_y) > dgig_quasi(cand, lambda, beta) / 2 - bound_y; // Check if U <= g(y) / unif(y)
273  } else {
274  rejected = true; // cand can be negative
275  }
276  }
277  res[i] = cand;
278  }
279 }

481 // Ratio-of-Uniforms with Mode Shift
482 inline void rgig_with_mode(Eigen::VectorXd& res, int num_sim, double lambda, double beta, boost::random::mt19937& rng) {
483  double mode = dgig_mode(lambda, beta); // argmax of g(x)
484  double quad_coef = -2 * (lambda + 1) / beta - mode;
485  double lin_coef = 2 * mode * (lambda - 1) / beta - 1;
486  double p = lin_coef - quad_coef * quad_coef / 3;
487  double q = 2 * quad_coef * quad_coef * quad_coef / 27 - quad_coef * lin_coef / 3 + mode;
488  double phi = acos(-q * sqrt(-27 / (p * p * p)) / 2);
489  double arg_x_neg = sqrt(-p * 4 / 3) * cos(phi / 3 + M_PI * 4 / 3) - quad_coef / 3;
490  double arg_x_pos = sqrt(-p * 4 / 3) * cos(phi / 3) - quad_coef / 3;
491  double bound_y = dgig_quasi(mode, lambda, beta) / 2; // use as normalize factor
492  double bound_x_neg = (arg_x_neg - mode) * exp(dgig_quasi(arg_x_neg, lambda, beta) / 2 - bound_y);
493  double bound_x_pos = (arg_x_pos - mode) * exp(dgig_quasi(arg_x_pos, lambda, beta) / 2 - bound_y);
494  bool rejected;
495  double draw_x, draw_y, cand; // bounded rectangle
496  for (int i = 0; i < num_sim; i++) {
497  rejected = true;
498  int iter_while = 0;
499  while (rejected && iter_while <= std::numeric_limits<int>::max() / 2) {
500  draw_x = (bound_x_pos - bound_x_neg) * unif_rand(rng) + bound_x_neg;
501  draw_y = unif_rand(rng); // U(0, 1) since g has been normalized
502  cand = draw_x / draw_y + mode;
503  if (cand > 0) {
504  rejected = log(draw_y) > dgig_quasi(cand, lambda, beta) / 2 - bound_y; // Check if U <= g(y) / unif(y)
505  } else {
506  rejected = true; // cand can be negative
507  }
508  ++iter_while;
509  }
510  res[i] = cand;
511  }
512 }
