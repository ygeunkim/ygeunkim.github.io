18inline Eigen::MatrixXd build_x0(const Eigen::MatrixXd& y, int var_lag, bool include_mean) {
19 int num_design = y.rows() - var_lag; // s = n - p
20 int dim = y.cols(); // m: dimension of the multivariate time series
21 int dim_design = dim * var_lag + 1; // k = mp + 1
22 Eigen::MatrixXd res(num_design, dim_design); // X0 = [Yp, ... Y1, 1]: s x k
23 for (int t = 0; t < var_lag; t++) {

80inline Eigen::MatrixXd build_ydummy(int p, const Eigen::VectorXd& sigma, double lambda,
81 const Eigen::VectorXd& daily, const Eigen::VectorXd& weekly, const Eigen::VectorXd& monthly,
82 bool include_mean) {
83 int dim = sigma.size();
84 Eigen::MatrixXd res = Eigen::MatrixXd::Zero(dim * p + dim + 1, dim); // Yp
85 // first block------------------------
86 res.block(0, 0, dim, dim).diagonal() = daily.array() * sigma.array() / lambda; // deltai * sigma or di * sigma
87 if (p > 1) {
88 // avoid error when p = 1
89 res.block(dim, 0, dim, dim).diagonal() = weekly.array() * sigma.array() / lambda; // wi * sigma
90 res.block(2 * dim, 0, dim, dim).diagonal() = monthly.array() * sigma.array() / lambda; // mi * sigma
91 }
92 // second block-----------------------
93 res.block(dim * p, 0, dim, dim).diagonal() = sigma;
94 if (include_mean) {
95 return res;
96 }
