83 inline double mgammafn(double x, int p) {
84  if (p < 1) {
85  STOP("'p' should be larger than or same as 1.");
86  }
87  if (x <= 0) {
88  STOP("'x' should be larger than 0.");
89  }
90  if (p == 1) {
91  return gammafn(x);
92  }
93  if (2 * x < p) {
94  STOP("'x / 2' should be larger than 'p'.");
95  }
96  double res = pow(M_PI, p * (p - 1) / 4.0);
97  for (int i = 0; i < p; i++) {
98  res *= gammafn(x - i / 2.0); // x + (1 - j) / 2
99  }
100  return res;
101 }
