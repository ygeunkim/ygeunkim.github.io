83inline double mgammafn(double x, int p) {
84 if (p < 1) {
85 STOP("'p' should be larger than or same as 1.");
86 }
87 if (x <= 0) {
88 STOP("'x' should be larger than 0.");
89 }
90 if (p == 1) {
91 return gammafn(x);
92 }
93 if (2 * x < p) {
94 STOP("'x / 2' should be larger than 'p'.");
95 }
96 double res = pow(M_PI, p * (p - 1) / 4.0);
97 for (int i = 0; i < p; i++) {
98 res *= gammafn(x - i / 2.0); // x + (1 - j) / 2
99 }
100 return res;
101}

138// Density of Inverse Gamma Distribution
139//
140// Compute the pdf of Inverse Gamma distribution
141//
142// @param x non-negative argument
143// @param shp Shape of the distribution
144// @param scl Scale of the distribution
145// @param lg If true, return log(f)
146inline double invgamma_dens(double x, double shp, double scl, bool lg) {
147 if (x < 0 ) {
148 STOP("'x' should be larger than 0.");
149 }
150 if (shp <= 0 ) {
151 STOP("'shp' should be larger than 0.");
152 }
153 if (scl <= 0 ) {
154 STOP("'scl' should be larger than 0.");
155 }
156 double res = pow(scl, shp) * pow(x, -shp - 1) * exp(-scl / x) / bvhar::gammafn(shp);
157 if (lg) {
158 return log(res);
159 }
160 return res;
161}
