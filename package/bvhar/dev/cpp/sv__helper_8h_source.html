17 double sv_sig, Eigen::Ref<Eigen::VectorXd> latent_vec, BHRNG& rng) {
18 int num_design = sv_vec.size(); // h_i1, ..., h_in for i = 1, .., k
19 // 7-component normal mixutre
20 using Vector7d = Eigen::Matrix<double, 7, 1>;
21 using Matrix7d = Eigen::Matrix<double, Eigen::Dynamic, 7>;
22 Vector7d pj(0.0073, 0.10556, 0.00002, 0.04395, 0.34001, 0.24566, 0.2575); // p_t
23 Vector7d muj(-10.12999, -3.97281, -8.56686, 2.77786, 0.61942, 1.79518, -1.08819); // mu_t
24 muj.array() -= 1.2704;

28 Eigen::VectorXd ds(num_design); // (mu_st - 1.2704)
29 Eigen::MatrixXd inv_sig_s = Eigen::MatrixXd::Zero(num_design, num_design); // diag(1 / sig_st^2)
30 Eigen::VectorXd inv_method(num_design); // inverse transform method
31 Matrix7d mixture_pdf(num_design, 7);
32 Matrix7d mixture_cumsum = Matrix7d::Zero(num_design, 7);
33 for (int i = 0; i < num_design; i++) {
34 inv_method[i] = unif_rand(0, 1, rng);
35 }
36 for (int i = 0; i < 7; i++) {
37 mixture_pdf.col(i) = (-((latent_vec.array() - sv_vec.array() - muj[i]).array() / sdj[i]).array().square() / 2).exp() * pj[i] / (sdj[i] * sqrt(2 * M_PI));
38 }
39 mixture_pdf.array().colwise() /= mixture_pdf.rowwise().sum().array();
40 for (int i = 0; i < 7; i++) {
41 mixture_cumsum.block(0, i, num_design, 7 - i) += mixture_pdf.col(i).rowwise().replicate(7 - i);
42 }
43 binom_latent.array() = 7 - (inv_method.rowwise().replicate(7).array() < mixture_cumsum.array()).cast<int>().rowwise().sum().array(); // 0 to 6 for indexing
44 Eigen::MatrixXd diff_mat = Eigen::MatrixXd::Identity(num_design, num_design);
45 for (int i = 0; i < num_design - 1; i++) {
46 ds[i] = muj[binom_latent[i]];
47 inv_sig_s(i, i) = 1 / sigj[binom_latent[i]];
48 diff_mat(i + 1, i) = -1;
49 }
50 ds[num_design - 1] = muj[binom_latent[num_design - 1]];
51 inv_sig_s(num_design - 1, num_design - 1) = 1 / sigj[binom_latent[num_design - 1]];
52 Eigen::MatrixXd HtH = diff_mat.transpose() * diff_mat;
53 Eigen::VectorXd res(num_design);
54 for (int i = 0; i < num_design; i++) {
55 res[i] = normal_rand(rng);
56 }
57 Eigen::MatrixXd post_sig = HtH / sv_sig + inv_sig_s;
58 Eigen::LLT<Eigen::MatrixXd> lltOfscale(post_sig);
59 Eigen::VectorXd post_mean = lltOfscale.solve(
60 HtH * init_sv * Eigen::VectorXd::Ones(num_design) / sv_sig + inv_sig_s * (latent_vec - ds)
61 );
62 sv_vec = post_mean + lltOfscale.matrixU().solve(res);
63}

74 Eigen::VectorXd& init_sv, Eigen::MatrixXd& h1, BHRNG& rng) {
75 int dim = init_sv.size();
76 int num_design = h1.rows();
77 Eigen::MatrixXd h_slide(num_design, dim); // h_ij, j = 0, ..., n - 1

79 h_slide.bottomRows(num_design - 1) = h1.topRows(num_design - 1);
