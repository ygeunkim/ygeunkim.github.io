152 void runGibbs(int chain) {
153 std::string log_name = fmt::format("Chain {}", chain + 1);
154 auto logger = spdlog::get(log_name);
155 if (logger == nullptr) {
156 logger = SPDLOG_SINK_MT(log_name);
157 }
158 logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
159 int logging_freq = num_iter / 20; // 5 percent
160 if (logging_freq == 0) {
161 logging_freq = 1;
162 }
163 bvharinterrupt();
164 for (int i = 0; i < num_burn; ++i) {
165 mcmc_ptr[chain]->doWarmUp();
166 if (display_progress && (i + 1) % logging_freq == 0) {
167 logger->info("{} / {} (Warmup)", i + 1, num_iter);
168 }
169 }
170 logger->flush();
171 for (int i = num_burn; i < num_iter; ++i) {
172 if (bvharinterrupt::is_interrupted()) {
173 logger->warn("User interrupt in {} / {}", i + 1, num_iter);
174 #ifdef _OPENMP
175 #pragma omp critical
176 #endif
177 {
178 res[chain] = mcmc_ptr[chain]->returnRecords(0, 1);
179 }
180 break;
181 }
182 mcmc_ptr[chain]->doPosteriorDraws();
183 if (display_progress && (i + 1) % logging_freq == 0) {
184 logger->info("{} / {} (Sampling)", i + 1, num_iter);
185 }
186 }
187 #ifdef _OPENMP
188 #pragma omp critical
189 #endif
190 {
191 res[chain] = mcmc_ptr[chain]->returnRecords(0, thin);
192 }
193 logger->flush();
194 spdlog::drop(log_name);
195 }
