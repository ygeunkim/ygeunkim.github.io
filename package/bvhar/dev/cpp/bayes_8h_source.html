132 void runGibbs(int chain) {
133 std::string log_name = fmt::format("Chain {}", chain + 1);
134 auto logger = spdlog::get(log_name);
135 if (logger == nullptr) {
136 logger = SPDLOG_SINK_MT(log_name);

138 logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
139 int logging_freq = num_iter / 20; // 5 percent
140 if (logging_freq == 0) {
141 logging_freq = 1;
142 }
143 bvharinterrupt();
144 for (int i = 0; i < num_burn; ++i) {
145 mcmc_ptr[chain]->doWarmUp();
146 if (display_progress && (i + 1) % logging_freq == 0) {
147 logger->info("{} / {} (Warmup)", i + 1, num_iter);
148 }
149 }
150 logger->flush();
151 for (int i = num_burn; i < num_iter; ++i) {
152 if (bvharinterrupt::is_interrupted()) {
153 logger->warn("User interrupt in {} / {}", i + 1, num_iter);
154 #ifdef _OPENMP
155 #pragma omp critical
156 #endif
157 {
158 res[chain] = mcmc_ptr[chain]->returnRecords(0, 1);
159 }
160 break;
161 }
162 mcmc_ptr[chain]->doPosteriorDraws();
163 if (display_progress && (i + 1) % logging_freq == 0) {
164 logger->info("{} / {} (Sampling)", i + 1, num_iter);
165 }
166 }
167 #ifdef _OPENMP

169 #endif
170 {
171 res[chain] = mcmc_ptr[chain]->returnRecords(0, thin);

173 logger->flush();
