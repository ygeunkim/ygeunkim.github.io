139 std::string log_name = fmt::format("Chain {}", chain + 1);
140 auto logger = spdlog::get(log_name);
141 if (logger == nullptr) {
142 logger = BVHAR_SPDLOG_SINK_MT(log_name);

144 logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
145 int logging_freq = num_iter / 20; // 5 percent
146 if (logging_freq == 0) {
147 logging_freq = 1;
148 }
149 BVHAR_INIT_DEBUG(logger);
150 bvharinterrupt();
151 for (int i = 0; i < num_burn; ++i) {
152 mcmc_ptr[chain]->doWarmUp();
153 BVHAR_DEBUG_LOG(logger, "{} / {} (Warmup)", i + 1, num_iter);
154 if (display_progress && (i + 1) % logging_freq == 0) {
155 logger->info("{} / {} (Warmup)", i + 1, num_iter);
156 }
157 }
158 logger->flush();
159 for (int i = num_burn; i < num_iter; ++i) {
160 if (bvharinterrupt::is_interrupted()) {
161 logger->warn("User interrupt in {} / {}", i + 1, num_iter);
162 #ifdef _OPENMP
163 #pragma omp critical
164 #endif
165 {
166 res[chain] = mcmc_ptr[chain]->returnRecords(0, 1);
167 }
168 break;
169 }
170 mcmc_ptr[chain]->doPosteriorDraws();
171 BVHAR_DEBUG_LOG(logger, "{} / {} (Sampling)", i + 1, num_iter);
172 if (display_progress && (i + 1) % logging_freq == 0) {
173 logger->info("{} / {} (Sampling)", i + 1, num_iter);
174 }
175 }
176 #ifdef _OPENMP

178 #endif
179 {
180 res[chain] = mcmc_ptr[chain]->returnRecords(0, thin);
