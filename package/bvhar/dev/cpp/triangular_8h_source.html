260 void updateCoef() {
261 for (int j = 0; j < dim; ++j) {
262 coef_mat.col(j).setZero(); // j-th column of A = 0
263 Eigen::MatrixXd chol_lower_j = chol_lower.bottomRows(dim - j); // L_(j:k) = a_jt to a_kt for t = 1, ..., j - 1
264 Eigen::MatrixXd sqrt_sv_j = sqrt_sv.rightCols(dim - j); // use h_jt to h_kt for t = 1, .. n => (k - j + 1) x k
265 Eigen::MatrixXd design_coef = kronecker_eigen(chol_lower_j.col(j), x).array().colwise() / sqrt_sv_j.reshaped().array(); // L_(j:k, j) otimes X0 scaled by D_(1:n, j:k): n(k - j + 1) x kp
266 Eigen::VectorXd prior_mean_j(dim_design);
267 Eigen::VectorXd prior_prec_j(dim_design);
268 Eigen::VectorXd penalty_j = Eigen::VectorXd::Zero(dim_design);
269 if (include_mean) {
270 prior_mean_j << prior_alpha_mean.segment(j * nrow_coef, nrow_coef), prior_alpha_mean.tail(dim)[j];
271 prior_prec_j << prior_alpha_prec.segment(j * nrow_coef, nrow_coef), prior_alpha_prec.tail(dim)[j];
272 // penalty_j << alpha_penalty.segment(j * nrow_coef, nrow_coef), alpha_penalty.tail(dim)[j];
273 penalty_j.head(nrow_coef) = alpha_penalty.segment(j * nrow_coef, nrow_coef);
274 draw_coef(
275 coef_mat.col(j), design_coef,
276 (((y - x * coef_mat) * chol_lower_j.transpose()).array() / sqrt_sv_j.array()).reshaped(), // Hadamard product between: (Y - X0 A(-j))L_(j:k)^T and D_(1:n, j:k)
277 prior_mean_j, prior_prec_j, rng
278 );
279 coef_vec.head(num_alpha) = coef_mat.topRows(nrow_coef).reshaped();
280 coef_vec.tail(dim) = coef_mat.bottomRows(1).transpose();
281 } else {
282 prior_mean_j = prior_alpha_mean.segment(dim_design * j, dim_design);
283 prior_prec_j = prior_alpha_prec.segment(dim_design * j, dim_design);
284 penalty_j = alpha_penalty.segment(dim_design * j, dim_design);
285 draw_coef(
286 coef_mat.col(j),
287 design_coef,
288 (((y - x * coef_mat) * chol_lower_j.transpose()).array() / sqrt_sv_j.array()).reshaped(),
289 prior_mean_j, prior_prec_j, rng
290 );
291 coef_vec = coef_mat.reshaped();
292 }
293 draw_mn_savs(sparse_coef.col(j), coef_mat.col(j), x, penalty_j);
294 }

301 void updateImpact() {

303 response_contem = latent_innov.col(j).array() / sqrt_sv.col(j).array(); // n-dim

306 draw_coef(

310 prior_chol_prec.segment(contem_id, j),
311 rng
312 );
313 draw_savs(sparse_contem.segment(contem_id, j), contem_coef.segment(contem_id, j), latent_innov.leftCols(j));

445template <typename BaseMcmc = McmcReg, bool isGroup = true>
446inline std::vector<std::unique_ptr<BaseMcmc>> initialize_mcmc(
447 int num_chains, int num_iter, const Eigen::MatrixXd& x, const Eigen::MatrixXd& y,
448 LIST& param_reg, LIST& param_prior, LIST& param_intercept, LIST_OF_LIST& param_init, int prior_type,
449 LIST& contem_prior, LIST_OF_LIST& contem_init, int contem_prior_type,
450 const Eigen::VectorXi& grp_id, const Eigen::VectorXi& own_id, const Eigen::VectorXi& cross_id, const Eigen::MatrixXi& grp_mat,
451 bool include_mean, Eigen::Ref<const Eigen::VectorXi> seed_chain, Optional<int> num_design = NULLOPT
452) {
453 using PARAMS = typename std::conditional<std::is_same<BaseMcmc, McmcReg>::value, RegParams, SvParams>::type;
454 using INITS = typename std::conditional<std::is_same<BaseMcmc, McmcReg>::value, LdltInits, SvInits>::type;
455 PARAMS base_params(
456 num_iter, x, y,
457 param_reg,
458 own_id, cross_id,
459 grp_id, grp_mat,
460 param_intercept, include_mean
461 );
462 std::vector<std::unique_ptr<BaseMcmc>> mcmc_ptr(num_chains);
463 for (int i = 0; i < num_chains; ++i) {
464 LIST init_spec = param_init[i];
465 auto coef_updater = initialize_shrinkageupdater<isGroup>(num_iter, param_prior, init_spec, prior_type);
466 coef_updater->initCoefMean(base_params._alpha_mean, base_params._num_alpha);
467 coef_updater->initCoefPrec(base_params._alpha_prec, base_params._num_alpha, base_params._grp_vec, base_params._cross_id);
468 LIST contem_init_spec = contem_init[i];
469 auto contem_updater = initialize_shrinkageupdater<isGroup>(num_iter, contem_prior, contem_init_spec, contem_prior_type);
470 contem_updater->initImpactPrec(base_params._chol_prec);
471 INITS ldlt_inits = num_design ? INITS(init_spec, *num_design) : INITS(init_spec);
472 mcmc_ptr[i] = std::make_unique<BaseMcmc>(
473 base_params, ldlt_inits,
474 coef_updater, contem_updater,
475 static_cast<unsigned int>(seed_chain[i])
476 );
477 }
478 return mcmc_ptr;
479}
