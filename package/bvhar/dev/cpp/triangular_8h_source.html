338 void updateImpact() {
339 for (int j = 1; j < dim; ++j) {
340 response_contem = latent_innov.col(j).array() / sqrt_sv.col(j).array(); // n-dim
341 Eigen::MatrixXd design_contem = latent_innov.leftCols(j).array().colwise() / sqrt_sv.col(j).reshaped().array(); // n x (j - 1)
342 contem_id = j * (j - 1) / 2;
343 draw_coef(
344 contem_coef.segment(contem_id, j),
345 design_contem, response_contem,
346 prior_chol_mean.segment(contem_id, j),
347 prior_chol_prec.segment(contem_id, j),
348 rng
349 );
350 draw_savs(sparse_contem.segment(contem_id, j), contem_coef.segment(contem_id, j), latent_innov.leftCols(j));
351 }
352 }

531 static_cast<unsigned int>(seed_chain[i]),
