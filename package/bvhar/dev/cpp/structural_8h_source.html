13inline Eigen::MatrixXd build_companion(Eigen::Ref<const Eigen::MatrixXd> coef_mat) {
14 int dim = coef_mat.cols();
15 int dim_design = coef_mat.rows();
16 Eigen::MatrixXd res = Eigen::MatrixXd::Zero(dim_design, dim_design);
17 res.topRows(dim) = coef_mat.transpose();
18 res.bottomLeftCorner(dim_design - dim, dim_design - dim).setIdentity();
19 return res;
20}

22inline Eigen::MatrixXd harx_to_var(Eigen::Ref<Eigen::MatrixXd> coef_mat, Eigen::Ref<Eigen::MatrixXd> har_trans) {
23 int dim = coef_mat.cols();
24 int dim_design = har_trans.cols();
25 int dim_exogen_design = coef_mat.rows() - har_trans.rows();
26 if (dim_exogen_design <= 0) {
27 return har_trans.transpose() * coef_mat;
28 }
29 int dim_har = coef_mat.rows() - dim_exogen_design;
30 Eigen::MatrixXd var_coef(dim_design + dim_exogen_design, dim); // rbind(C^T Phi, B)
31 var_coef.topRows(dim_design) = har_trans.transpose() * coef_mat.topRows(dim_har);
32 var_coef.bottomRows(dim_exogen_design) = coef_mat.bottomRows(dim_exogen_design);
33 return var_coef;
34}

59inline Eigen::MatrixXd convert_var_to_vma(Eigen::Ref<Eigen::MatrixXd> var_coef, int var_lag, int lag_max) {
60 int dim = var_coef.cols(); // m
61 if (lag_max < 1) {
62 BVHAR_STOP("'lag_max' must larger than 0");
63 }
64 int ma_rows = dim * (lag_max + 1);
65 int num_full_arows = ma_rows;
66 if (lag_max < var_lag) {
67 num_full_arows = dim * var_lag; // for VMA coefficient q < VAR(p)
68 }
69 Eigen::MatrixXd FullA = Eigen::MatrixXd::Zero(num_full_arows, dim); // same size with VMA coefficient matrix
70 FullA.topRows(dim * var_lag) = var_coef.topRows(dim * var_lag); // fill first mp row with VAR coefficient matrix
71 // Eigen::MatrixXd Im = Eigen::MatrixXd::Identity(dim, dim); // identity matrix
72 Eigen::MatrixXd ma = Eigen::MatrixXd::Zero(ma_rows, dim); // VMA [W1^T, W2^T, ..., W(lag_max)^T]^T, ma_rows = m * lag_max
73 ma.topRows(dim) = Eigen::MatrixXd::Identity(dim, dim); // W0 = Im
74 ma.middleRows(dim, dim) = FullA.topRows(dim) * ma.topRows(dim); // W1^T = B1^T * W1^T
75 if (lag_max == 1) {
76 return ma;
77 }
78 for (int i = 2; i < (lag_max + 1); i++) { // from W2: m-th row
79 for (int k = 0; k < i; k++) {
80 ma.middleRows(i * dim, dim) += FullA.middleRows(k * dim, dim) * ma.middleRows((i - k - 1) * dim, dim); // Wi = sum(W(i - k)^T * Bk^T)
81 }
82 }
83 return ma;
84}

86inline Eigen::MatrixXd convert_vma_ortho(Eigen::Ref<Eigen::MatrixXd> var_coef, Eigen::Ref<Eigen::MatrixXd> var_covmat, int var_lag, int lag_max) {
87 int dim = var_covmat.cols(); // num_rows = num_cols
88 if ((dim != var_covmat.rows()) && (dim != var_coef.cols())) {
89 BVHAR_STOP("Wrong covariance matrix format: `var_covmat`.");
90 }
91 if ((var_coef.rows() != var_lag * dim + 1) && (var_coef.rows() != var_lag * dim)) {
92 BVHAR_STOP("Wrong VAR coefficient format: `var_coef`.");
93 }
94 Eigen::MatrixXd ma = convert_var_to_vma(var_coef, var_lag, lag_max);
95 Eigen::MatrixXd res(ma.rows(), dim);
96 Eigen::LLT<Eigen::MatrixXd> lltOfcovmat(Eigen::Map<Eigen::MatrixXd>(var_covmat.data(), dim, dim)); // cholesky decomposition for Sigma
97 Eigen::MatrixXd chol_covmat = lltOfcovmat.matrixU();
98 for (int i = 0; i < lag_max + 1; i++) {
99 res.middleRows(i * dim, dim) = chol_covmat * ma.middleRows(i * dim, dim);
100 }
101 return res;
102}

109 Eigen::MatrixXd mse = Eigen::MatrixXd::Zero(dim * step, dim);

175inline Eigen::MatrixXd compute_vma_fevd(Eigen::Ref<Eigen::MatrixXd> vma_coef, Eigen::Ref<Eigen::MatrixXd> cov_mat, bool normalize) {
176 int dim = cov_mat.cols();
177 // Eigen::MatrixXd vma_mat = VARcoeftoVMA(var_coef, var_lag, step);
178 int step = vma_coef.rows() / dim; // h-step
179 Eigen::MatrixXd innov_account = Eigen::MatrixXd::Zero(dim, dim);
180 Eigen::MatrixXd ma_prod(dim, dim);
181 Eigen::MatrixXd numer = Eigen::MatrixXd::Zero(dim, dim);
182 Eigen::MatrixXd denom = Eigen::MatrixXd::Zero(dim, dim);
183 Eigen::MatrixXd res = Eigen::MatrixXd::Zero(dim * step, dim);
184 Eigen::MatrixXd cov_diag = Eigen::MatrixXd::Zero(dim, dim);
185 cov_diag.diagonal() = 1 / cov_mat.diagonal().cwiseSqrt().array(); // sigma_jj
186 for (int i = 0; i < step; i++) {
187 ma_prod = vma_coef.block(i * dim, 0, dim, dim).transpose() * cov_mat; // A * Sigma
188 innov_account += ma_prod * vma_coef.block(i * dim, 0, dim, dim); // A * Sigma * A^T
189 numer.array() += (ma_prod * cov_diag).array().square(); // sum(A * Sigma)_ij / sigma_jj^2
190 denom.diagonal() = 1 / innov_account.diagonal().array(); // sigma_jj^(-1) / sum(A * Sigma * A^T)_jj
191 res.block(i * dim, 0, dim, dim) = denom * numer; // sigma_jj^(-1) sum(A * Sigma)_ij / sum(A * Sigma * A^T)_jj
192 }
193 if (normalize) {
194 res.array().colwise() /= res.rowwise().sum().array();
195 }
196 return res;
197}
