593template <bool isGroup = true>
594inline std::unique_ptr<ShrinkageUpdater> initialize_shrinkageupdater(int num_iter, LIST& param_prior, LIST& param_init, int prior_type) {
595 std::unique_ptr<ShrinkageUpdater> shrinkage_ptr;
596 switch (prior_type) {
597 case 1: {
598 std::unique_ptr<MinnParams> params_ptr;
599 if (CONTAINS(param_prior, "p")) {
600 // p is only in coef_prior
601 params_ptr = std::make_unique<MinnParams>(param_prior);
602 } else {
603 // append num_lowerchol to param_prior when contem
604 params_ptr = std::make_unique<MinnParams>(param_prior, CAST_INT(param_prior["num"]));
605 }
606 ShrinkageInits inits(param_init);
607 shrinkage_ptr = std::make_unique<MinnUpdater>(num_iter, *params_ptr, inits);
608 return shrinkage_ptr;
609 }
610 case 2: {
611 SsvsParams params(param_prior);
612 SsvsInits inits(param_init);
613 shrinkage_ptr = std::make_unique<SsvsUpdater>(num_iter, params, inits);
614 return shrinkage_ptr;
615 }
616 case 3: {
617 ShrinkageParams params(param_prior);
