577template <bool isGroup = true>
578inline std::unique_ptr<ShrinkageUpdater> initialize_shrinkageupdater(int num_iter, BVHAR_LIST& param_prior, BVHAR_LIST& param_init, int prior_type) {
579 std::unique_ptr<ShrinkageUpdater> shrinkage_ptr;
580 switch (prior_type) {
581 case 1: {
582 std::unique_ptr<MinnParams> params_ptr;
583 if (BVHAR_CONTAINS(param_prior, "p")) {
584 // p is only in coef_prior
585 params_ptr = std::make_unique<MinnParams>(param_prior);
586 } else {
587 // append num_lowerchol to param_prior when contem
588 params_ptr = std::make_unique<MinnParams>(param_prior, BVHAR_CAST_INT(param_prior["num"]));
589 }
590 ShrinkageInits inits(param_init);
591 shrinkage_ptr = std::make_unique<MinnUpdater>(num_iter, *params_ptr, inits);
592 return shrinkage_ptr;
593 }
594 case 2: {
595 SsvsParams params(param_prior);
596 SsvsInits inits(param_init);
597 shrinkage_ptr = std::make_unique<SsvsUpdater>(num_iter, params, inits);
598 return shrinkage_ptr;
599 }
600 case 3: {
601 ShrinkageParams params(param_prior);
602 HorseshoeInits inits(param_init);
603 shrinkage_ptr = std::make_unique<HorseshoeUpdater<isGroup>>(num_iter, params, inits);
604 return shrinkage_ptr;
605 }
606 case 4: {
607 std::unique_ptr<HierminnParams> params_ptr;
608 if (BVHAR_CONTAINS(param_prior, "p")) {
609 params_ptr = std::make_unique<HierminnParams>(param_prior);
610 } else {
611 params_ptr = std::make_unique<HierminnParams>(param_prior, BVHAR_CAST_INT(param_prior["num"]));
