578template <bool isGroup = true>
579inline std::unique_ptr<ShrinkageUpdater> initialize_shrinkageupdater(int num_iter, BVHAR_LIST& param_prior, BVHAR_LIST& param_init, int prior_type) {
580 std::unique_ptr<ShrinkageUpdater> shrinkage_ptr;
581 switch (prior_type) {
582 case 1: {
583 std::unique_ptr<MinnParams> params_ptr;
584 if (BVHAR_CONTAINS(param_prior, "p")) {
585 // p is only in coef_prior
586 params_ptr = std::make_unique<MinnParams>(param_prior);
587 } else {
588 // append num_lowerchol to param_prior when contem
589 params_ptr = std::make_unique<MinnParams>(param_prior, BVHAR_CAST_INT(param_prior["num"]));
590 }
591 ShrinkageInits inits(param_init);
592 shrinkage_ptr = std::make_unique<MinnUpdater>(num_iter, *params_ptr, inits);
593 return shrinkage_ptr;
594 }
595 case 2: {
596 SsvsParams params(param_prior);
597 SsvsInits inits(param_init);
598 shrinkage_ptr = std::make_unique<SsvsUpdater>(num_iter, params, inits);
599 return shrinkage_ptr;
600 }
601 case 3: {
602 ShrinkageParams params(param_prior);
603 HorseshoeInits inits(param_init);
604 shrinkage_ptr = std::make_unique<HorseshoeUpdater<isGroup>>(num_iter, params, inits);
605 return shrinkage_ptr;
606 }
607 case 4: {
608 std::unique_ptr<HierminnParams> params_ptr;
609 if (BVHAR_CONTAINS(param_prior, "p")) {
610 params_ptr = std::make_unique<HierminnParams>(param_prior);
611 } else {
612 params_ptr = std::make_unique<HierminnParams>(param_prior, BVHAR_CAST_INT(param_prior["num"]));
