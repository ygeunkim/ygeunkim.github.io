395 void runGibbs(int window, int chain) {
396 BVHAR_DEBUG_LOG(debug_logger, "runGibbs(window={}, chain={}) called", window, chain);
397 std::string log_name = fmt::format("Chain {} / Window {}", chain + 1, window + 1);
398 auto logger = spdlog::get(log_name);
399 if (logger == nullptr) {
400 logger = BVHAR_SPDLOG_SINK_MT(log_name);
401 }
402 logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
403 int logging_freq = num_iter / 20; // 5 percent
404 if (logging_freq == 0) {
405 logging_freq = 1;
406 }
407 BVHAR_INIT_DEBUG(logger);
408 bvharinterrupt();
409 for (int i = 0; i < num_burn; ++i) {
410 model[window][chain]->doWarmUp();
411 BVHAR_DEBUG_LOG(logger, "{} / {} (Warmup)", i + 1, num_iter);
412 if (display_progress && (i + 1) % logging_freq == 0) {
413 logger->info("{} / {} (Warmup)", i + 1, num_iter);
414 }
415 }
416 logger->flush();
417 for (int i = num_burn; i < num_iter; ++i) {
418 if (bvharinterrupt::is_interrupted()) {
419 logger->warn("User interrupt in {} / {}", i + 1, num_iter);
420 break;
421 }
422 model[window][chain]->doPosteriorDraws();
423 BVHAR_DEBUG_LOG(logger, "{} / {} (Sampling)", i + 1, num_iter);
424 if (display_progress && (i + 1) % logging_freq == 0) {
425 logger->info("{} / {} (Sampling)", i + 1, num_iter);
426 }
427 }
428 // RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
429 // updateForecaster(reg_record, window, chain);
430 // model[window][chain].reset();
431 updateForecaster(window, chain);
432 logger->flush();
433 spdlog::drop(log_name);
