100inline Eigen::MatrixXd build_ydummy(int p, const Eigen::VectorXd& sigma, double lambda,
101 const Eigen::VectorXd& daily, const Eigen::VectorXd& weekly, const Eigen::VectorXd& monthly,
102 bool include_mean) {
103 int dim = sigma.size();
104 Eigen::MatrixXd res = Eigen::MatrixXd::Zero(dim * p + dim + 1, dim); // Yp
105 // first block------------------------
106 res.block(0, 0, dim, dim).diagonal() = daily.array() * sigma.array() / lambda; // deltai * sigma or di * sigma
107 if (p > 1) {
108 // avoid error when p = 1
109 res.block(dim, 0, dim, dim).diagonal() = weekly.array() * sigma.array() / lambda; // wi * sigma
110 res.block(2 * dim, 0, dim, dim).diagonal() = monthly.array() * sigma.array() / lambda; // mi * sigma
111 }
112 // second block-----------------------
113 res.block(dim * p, 0, dim, dim).diagonal() = sigma;
114 if (include_mean) {
115 return res;
116 }
117 return res.topRows(dim * p + dim);
118}
