247 void updateCoef() {
248 for (int j = 0; j < dim; ++j) {
249 coef_mat.col(j).setZero(); // j-th column of A = 0
250 Eigen::MatrixXd chol_lower_j = chol_lower.bottomRows(dim - j); // L_(j:k) = a_jt to a_kt for t = 1, ..., j - 1
251 Eigen::MatrixXd sqrt_sv_j = sqrt_sv.rightCols(dim - j); // use h_jt to h_kt for t = 1, .. n => (k - j + 1) x k
252 Eigen::MatrixXd design_coef = kronecker_eigen(chol_lower_j.col(j), x).array().colwise() / sqrt_sv_j.reshaped().array(); // L_(j:k, j) otimes X0 scaled by D_(1:n, j:k): n(k - j + 1) x kp
253 Eigen::VectorXd prior_mean_j(dim_design);
254 Eigen::VectorXd prior_prec_j(dim_design);
255 Eigen::VectorXd penalty_j = Eigen::VectorXd::Zero(dim_design);
256 if (include_mean) {
257 prior_mean_j << prior_alpha_mean.segment(j * nrow_coef, nrow_coef), prior_alpha_mean.tail(dim)[j];
258 prior_prec_j << prior_alpha_prec.segment(j * nrow_coef, nrow_coef), prior_alpha_prec.tail(dim)[j];
259 // penalty_j << alpha_penalty.segment(j * nrow_coef, nrow_coef), alpha_penalty.tail(dim)[j];
260 penalty_j.head(nrow_coef) = alpha_penalty.segment(j * nrow_coef, nrow_coef);
261 draw_coef(
262 coef_mat.col(j), design_coef,
263 (((y - x * coef_mat) * chol_lower_j.transpose()).array() / sqrt_sv_j.array()).reshaped(), // Hadamard product between: (Y - X0 A(-j))L_(j:k)^T and D_(1:n, j:k)
264 prior_mean_j, prior_prec_j, rng
265 );
266 coef_vec.head(num_alpha) = coef_mat.topRows(nrow_coef).reshaped();
267 coef_vec.tail(dim) = coef_mat.bottomRows(1).transpose();
268 } else {
269 prior_mean_j = prior_alpha_mean.segment(dim_design * j, dim_design);
270 prior_prec_j = prior_alpha_prec.segment(dim_design * j, dim_design);
271 penalty_j = alpha_penalty.segment(dim_design * j, dim_design);
272 draw_coef(
273 coef_mat.col(j),
274 design_coef,
275 (((y - x * coef_mat) * chol_lower_j.transpose()).array() / sqrt_sv_j.array()).reshaped(),
276 prior_mean_j, prior_prec_j, rng
277 );
278 coef_vec = coef_mat.reshaped();
279 }
280 draw_mn_savs(sparse_coef.col(j), coef_mat.col(j), x, penalty_j);
281 }
282 }

288 void updateImpact() {
289 for (int j = 1; j < dim; ++j) {
290 response_contem = latent_innov.col(j).array() / sqrt_sv.col(j).array(); // n-dim
291 Eigen::MatrixXd design_contem = latent_innov.leftCols(j).array().colwise() / sqrt_sv.col(j).reshaped().array(); // n x (j - 1)
292 contem_id = j * (j - 1) / 2;
293 draw_coef(
294 contem_coef.segment(contem_id, j),
295 design_contem, response_contem,
296 prior_chol_mean.segment(contem_id, j),
297 prior_chol_prec.segment(contem_id, j),
298 rng
299 );
300 draw_savs(sparse_contem.segment(contem_id, j), contem_coef.segment(contem_id, j), latent_innov.leftCols(j));
301 }
