775 void runGibbs(int window, int chain) {
776 std::string log_name = fmt::format("Chain {} / Window {}", chain + 1, window + 1);
777 auto logger = spdlog::get(log_name);
778 if (logger == nullptr) {
779 logger = SPDLOG_SINK_MT(log_name);
780 }
781 logger->set_pattern("[%n] [Thread " + std::to_string(omp_get_thread_num()) + "] %v");
782 int logging_freq = num_iter / 20; // 5 percent
783 if (logging_freq == 0) {
784 logging_freq = 1;
785 }
786 bvharinterrupt();
787 for (int i = 0; i < num_burn; ++i) {
788 model[window][chain]->doWarmUp();
789 if (display_progress && (i + 1) % logging_freq == 0) {
790 logger->info("{} / {} (Warmup)", i + 1, num_iter);
791 }
792 }
793 logger->flush();
794 for (int i = num_burn; i < num_iter; ++i) {
795 if (bvharinterrupt::is_interrupted()) {
796 RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
797 logger->warn("User interrupt in {} / {}", i + 1, num_iter);
798 break;
799 }
800 model[window][chain]->doPosteriorDraws();
801 if (display_progress && (i + 1) % logging_freq == 0) {
802 logger->info("{} / {} (Sampling)", i + 1, num_iter);
803 }
804 }
805 RecordType reg_record = model[window][chain]->template returnStructRecords<RecordType>(0, thin, sparse);
806 updateForecaster(reg_record, window, chain);
807 model[window][chain].reset();
808 logger->flush();
809 spdlog::drop(log_name);
810 }

819 using is_mcmc = std::integral_constant<bool, isUpdate>;
820 if (window != 0 && is_mcmc::value) {
821 runGibbs(window, chain);
822 }
823 Eigen::VectorXd valid_vec = y_test.row(step);
824 out_forecast[window][chain] = forecaster[window][chain]->forecastDensity(valid_vec).bottomRows(1);
825 lpl_record(window, chain) = forecaster[window][chain]->returnLpl();
826 forecaster[window][chain].reset(); // free the memory by making nullpt
827 }
