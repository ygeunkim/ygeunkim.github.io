40struct MinnParams : public ShrinkageParams {
41 // Eigen::MatrixXd _prec_diag;
42 Eigen::VectorXd _prior_prec, _prior_mean;
43
44 MinnParams(LIST& priors)
45 : ShrinkageParams(priors) {
46 int lag = CAST_INT(priors["p"]); // append to bayes_spec, p = 3 in VHAR
47 Eigen::VectorXd sigma = CAST<Eigen::VectorXd>(priors["sigma"]);
48 // double lam = CAST_DOUBLE(priors["lambda"]);
49 double lam;
50 if (CAST_BOOL(priors["hierarchical"])) {
51 lam = 1; // when Hierminn
52 } else {
53 lam = CAST_DOUBLE(priors["lambda"]); // when Minn
54 }
55 double eps = CAST_DOUBLE(priors["eps"]);
56 int dim = sigma.size();
57 Eigen::MatrixXd prec_diag = Eigen::MatrixXd::Zero(dim, dim);
58 Eigen::VectorXd daily(dim);
59 Eigen::VectorXd weekly(dim);
60 Eigen::VectorXd monthly(dim);
61 if (CONTAINS(priors, "delta")) {
62 daily = CAST<Eigen::VectorXd>(priors["delta"]);
63 weekly.setZero();
64 monthly.setZero();
65 } else {
66 daily = CAST<Eigen::VectorXd>(priors["daily"]);
67 weekly = CAST<Eigen::VectorXd>(priors["weekly"]);
68 monthly = CAST<Eigen::VectorXd>(priors["monthly"]);
69 }
70 Eigen::MatrixXd dummy_response = build_ydummy(lag, sigma, lam, daily, weekly, monthly, false);
71 Eigen::MatrixXd dummy_design = build_xdummy(
72 Eigen::VectorXd::LinSpaced(lag, 1, lag),
73 lam, sigma, eps, false
74 );
75 Eigen::MatrixXd prior_prec = dummy_design.transpose() * dummy_design;
76 _prior_mean = prior_prec.llt().solve(dummy_design.transpose() * dummy_response).reshaped();
77 prec_diag.diagonal() = 1 / sigma.array();
78 _prior_prec = kronecker_eigen(prec_diag, prior_prec).diagonal();
79 }
80
81 MinnParams(LIST& priors, int num_lowerchol)
82 : ShrinkageParams(priors),
83 _prior_prec(Eigen::VectorXd::Ones(num_lowerchol)),
84 _prior_mean(Eigen::VectorXd::Zero(num_lowerchol)) {}
85};
