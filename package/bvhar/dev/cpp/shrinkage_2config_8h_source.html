41struct MinnParams : public ShrinkageParams {
42 // Eigen::MatrixXd _prec_diag;
43 Eigen::VectorXd _prior_prec, _prior_mean;
44
45 MinnParams(BVHAR_LIST& priors)
46 : ShrinkageParams(priors) {
47 int lag = BVHAR_CAST_INT(priors["p"]); // append to bayes_spec, p = 3 in VHAR
48 Eigen::VectorXd sigma = BVHAR_CAST<Eigen::VectorXd>(priors["sigma"]);
49 // double lam = BVHAR_CAST_DOUBLE(priors["lambda"]);
50 double lam;
51 if (BVHAR_CAST_BOOL(priors["hierarchical"])) {
52 lam = 1; // when Hierminn
53 } else {
54 lam = BVHAR_CAST_DOUBLE(priors["lambda"]); // when Minn
55 }
56 double eps = BVHAR_CAST_DOUBLE(priors["eps"]);
57 int dim = sigma.size();
58 Eigen::MatrixXd prec_diag = Eigen::MatrixXd::Zero(dim, dim);
59 Eigen::VectorXd daily(dim);
60 Eigen::VectorXd weekly(dim);
61 Eigen::VectorXd monthly(dim);
62 if (BVHAR_CONTAINS(priors, "delta")) {
63 daily = BVHAR_CAST<Eigen::VectorXd>(priors["delta"]);
64 weekly.setZero();
65 monthly.setZero();
66 } else {
67 daily = BVHAR_CAST<Eigen::VectorXd>(priors["daily"]);
68 weekly = BVHAR_CAST<Eigen::VectorXd>(priors["weekly"]);
69 monthly = BVHAR_CAST<Eigen::VectorXd>(priors["monthly"]);
70 }
71 Eigen::MatrixXd dummy_response = build_ydummy(lag, sigma, lam, daily, weekly, monthly, false);
72 Eigen::MatrixXd dummy_design = build_xdummy(
73 Eigen::VectorXd::LinSpaced(lag, 1, lag),
74 lam, sigma, eps, false
75 );
76 Eigen::MatrixXd prior_prec = dummy_design.transpose() * dummy_design;
77 _prior_mean = prior_prec.llt().solve(dummy_design.transpose() * dummy_response).reshaped();
78 prec_diag.diagonal() = 1 / sigma.array();
79 _prior_prec = kronecker_eigen(prec_diag, prior_prec).diagonal();
80 }
81
82 MinnParams(BVHAR_LIST& priors, int num_lowerchol)
83 : ShrinkageParams(priors),
84 _prior_prec(Eigen::VectorXd::Ones(num_lowerchol)),
85 _prior_mean(Eigen::VectorXd::Zero(num_lowerchol)) {}
86};
