50class MultiOls {
51public:
52 MultiOls(const Eigen::MatrixXd& x, const Eigen::MatrixXd& y)
53 : design(x), response(y),
54 dim(response.cols()), num_design(response.rows()), dim_design(design.cols()) {
55 coef = Eigen::MatrixXd::Zero(dim_design, dim);
56 yhat = Eigen::MatrixXd::Zero(num_design, dim);
57 resid = Eigen::MatrixXd::Zero(num_design, dim);
58 cov = Eigen::MatrixXd::Zero(dim, dim);
59 }
60 virtual ~MultiOls() = default;
61 virtual void estimateCoef() {
62 coef = (design.transpose() * design).inverse() * design.transpose() * response; // return coef -> use in OlsVar
63 }
64 virtual void fitObs() {
65 yhat = design * coef;
66 resid = response - yhat;
67 }
68 void estimateCov() {
69 cov = resid.transpose() * resid / (num_design - dim_design);
70 }
71 BVHAR_LIST returnOlsRes() {
72 estimateCoef();
73 fitObs();
74 estimateCov();
75 return BVHAR_CREATE_LIST(
76 BVHAR_NAMED("coefficients") = coef,
77 BVHAR_NAMED("fitted.values") = yhat,
78 BVHAR_NAMED("residuals") = resid,
79 BVHAR_NAMED("covmat") = cov,
80 BVHAR_NAMED("df") = dim_design,
81 BVHAR_NAMED("m") = dim,
82 BVHAR_NAMED("obs") = num_design,
83 BVHAR_NAMED("y0") = response
84 );
85 }
86 Eigen::MatrixXd returnCoef() {
87 estimateCoef();
88 fitObs();
89 estimateCov();
90 return coef;
91 }
92 OlsFit returnOlsFit(int ord) {
