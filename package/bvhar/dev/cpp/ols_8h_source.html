51class MultiOls {
52public:
53 MultiOls(const Eigen::MatrixXd& x, const Eigen::MatrixXd& y)
54 : design(x), response(y),
55 dim(response.cols()), num_design(response.rows()), dim_design(design.cols()) {
56 coef = Eigen::MatrixXd::Zero(dim_design, dim);
57 yhat = Eigen::MatrixXd::Zero(num_design, dim);
58 resid = Eigen::MatrixXd::Zero(num_design, dim);
59 cov = Eigen::MatrixXd::Zero(dim, dim);
60 }
61 virtual ~MultiOls() = default;
62 virtual void estimateCoef() {
63 coef = (design.transpose() * design).inverse() * design.transpose() * response; // return coef -> use in OlsVar
64 }
65 virtual void fitObs() {
66 yhat = design * coef;
67 resid = response - yhat;
68 }
69 void estimateCov() {
70 cov = resid.transpose() * resid / (num_design - dim_design);
71 }
72 BVHAR_LIST returnOlsRes() {
73 estimateCoef();
74 fitObs();
75 estimateCov();
76 return BVHAR_CREATE_LIST(
77 BVHAR_NAMED("coefficients") = coef,
78 BVHAR_NAMED("fitted.values") = yhat,
79 BVHAR_NAMED("residuals") = resid,
80 BVHAR_NAMED("covmat") = cov,
81 BVHAR_NAMED("df") = dim_design,
82 BVHAR_NAMED("m") = dim,
83 BVHAR_NAMED("obs") = num_design,
84 BVHAR_NAMED("y0") = response
85 );
86 }
87 Eigen::MatrixXd returnCoef() {
88 estimateCoef();
89 fitObs();
90 estimateCov();
91 return coef;
92 }
93 OlsFit returnOlsFit(int ord) {

154inline std::unique_ptr<MultiOls> initialize_ols(const Eigen::MatrixXd& design, const Eigen::MatrixXd& response, int method) {
155 std::unique_ptr<MultiOls> ols_ptr;
156 switch (method) {
