8// Building Spike-and-slab SD Diagonal Matrix
9//
10// In MCMC process of SSVS, this function computes diagonal matrix \eqn{D} or \eqn{D_j} defined by spike-and-slab sd.
11//
12// @param spike_sd Standard deviance for Spike normal distribution
13// @param slab_sd Standard deviance for Slab normal distribution
14// @param mixture_dummy Indicator vector (0-1) corresponding to each element
15inline Eigen::VectorXd build_ssvs_sd(Eigen::VectorXd spike_sd, Eigen::VectorXd slab_sd, Eigen::VectorXd mixture_dummy) {
16 Eigen::VectorXd res(spike_sd.size());
17 res.array() = (1 - mixture_dummy.array()) * spike_sd.array() + mixture_dummy.array() * slab_sd.array(); // diagonal term = spike_sd if mixture_dummy = 0 while slab_sd if mixture_dummy = 1
18 return res;

21// Generating Dummy Vector for Parameters in SSVS Gibbs Sampler
22//
23// In MCMC process of SSVS, this function generates latent \eqn{\gamma_j} or \eqn{\omega_{ij}} conditional posterior.
24//
25// @param param_obs Realized parameters vector
26// @param sd_numer Standard deviance for Slab normal distribution, which will be used for numerator.
27// @param sd_denom Standard deviance for Spike normal distribution, which will be used for denominator.
28// @param slab_weight Proportion of nonzero coefficients
29inline void ssvs_dummy(Eigen::VectorXd& dummy, Eigen::VectorXd param_obs,
30 Eigen::VectorXd& sd_numer, Eigen::Ref<const Eigen::VectorXd> sd_denom,
31 Eigen::VectorXd& slab_weight, BHRNG& rng) {
32 int num_latent = slab_weight.size();
33 Eigen::VectorXd exp_u1 = -param_obs.array().square() / (2 * sd_numer.array().square());
34 Eigen::VectorXd exp_u2 = -param_obs.array().square() / (2 * sd_denom.array().square());
35 Eigen::VectorXd max_exp = exp_u1.cwiseMax(exp_u2); // use log-sum-exp against overflow
36 exp_u1 = slab_weight.array() * (exp_u1 - max_exp).array().exp() / sd_numer.array();
37 exp_u2 = (1 - slab_weight.array()) * (exp_u2 - max_exp).array().exp() / sd_denom.array();
38 for (int i = 0; i < num_latent; i++) {
39 dummy[i] = ber_rand(exp_u1[i] / (exp_u1[i] + exp_u2[i]), rng);
40 }
41}

43// Generating Slab Weight Vector in SSVS Gibbs Sampler
44//
45// In MCMC process of SSVS, this function generates \eqn{p_j}.
46//
47// @param param_obs Indicator variables
48// @param prior_s1 First prior shape of Beta distribution
49// @param prior_s2 Second prior shape of Beta distribution
50inline void ssvs_weight(Eigen::VectorXd& weight, Eigen::VectorXd param_obs, double prior_s1, double prior_s2, BHRNG& rng) {
51 int num_latent = param_obs.size();
52 double post_s1 = prior_s1 + param_obs.sum(); // s1 + number of ones
53 double post_s2 = prior_s2 + num_latent - param_obs.sum(); // s2 + number of zeros
54 for (int i = 0; i < num_latent; i++) {
55 weight[i] = beta_rand(post_s1, post_s2, rng);
56 }
57}

59// Generating Slab Weight Vector in MN-SSVS Gibbs Sampler
60//
61// In MCMC process of SSVS, this function generates \eqn{p_j}.
62//
63// @param grp_vec Group vector
64// @param grp_id Unique group id
65// @param param_obs Indicator variables
66// @param prior_s1 First prior shape of Beta distribution
67// @param prior_s2 Second prior shape of Beta distribution
68inline void ssvs_mn_weight(Eigen::VectorXd& weight, Eigen::VectorXi& grp_vec, Eigen::VectorXi& grp_id,
69 Eigen::VectorXd& param_obs, Eigen::VectorXd& prior_s1, Eigen::VectorXd& prior_s2, BHRNG& rng) {
70 int num_grp = grp_id.size();
71 int num_latent = param_obs.size();
72 Eigen::Array<bool, Eigen::Dynamic, 1> global_id;
73 int mn_size = 0;
74 for (int i = 0; i < num_grp; i++) {
75 global_id = grp_vec.array() == grp_id[i];
76 mn_size = global_id.count();
77 Eigen::VectorXd mn_param(mn_size);
78 for (int j = 0, k = 0; j < num_latent; ++j) {
79 if (global_id[j]) {
80 mn_param[k++] = param_obs[j];
81 }
82 }
83 weight[i] = beta_rand(prior_s1[i] + mn_param.sum(), prior_s2[i] + mn_size - mn_param.sum(), rng);
84 }
85}

87// Generating SSVS Local Slab Parameter
88//
89// @param slab_param Slab parameter
90// @param dummy_param Bernoulli parameter
91// @param coef_vec Coefficient
92// @param shp IG shape for slab parameter
93// @param scl IG scale for slab parameter
94// @param spike_scl scaling factor to make spike sd smaller than slab sd (spike_sd = spike_scl * slab_sd)
95// @param rng boost rng
96inline void ssvs_local_slab(Eigen::VectorXd& slab_param, Eigen::VectorXd& dummy_param, Eigen::Ref<Eigen::VectorXd> coef_vec,
97 double& shp, double& scl, double& spike_scl, BHRNG& rng) {
98 for (int i = 0; i < coef_vec.size(); ++i) {
99 slab_param[i] = sqrt(1 / gamma_rand(
100 shp + .5,
101 1 / (scl + coef_vec[i] * coef_vec[i] / (dummy_param[i] + (1 - dummy_param[i]) * spike_scl)),
102 rng
103 ));
104 }
