44 void computeSpillover() {
45 for (int i = 0; i < num_sim; ++i) {
46 reg_record->updateDiag(i, time_id, sv_update);
47 sqrt_sig = build_inv_lower(
48 dim,
49 reg_record->contem_coef_record.row(i)
50 ).triangularView<Eigen::UnitLower>().solve(sv_update.asDiagonal().toDenseMatrix());
51 cov = sqrt_sig * sqrt_sig.transpose();
52 coef_mat = unvectorize(reg_record->coef_record.row(i).transpose(), dim);
53 computeVma();
54 fevd.middleCols(i * dim, dim) = compute_vma_fevd(vma_mat, cov, true);
55 spillover.middleCols(i * dim, dim) = compute_sp_index(fevd.middleCols(i * dim, dim));
56 to_spillover.segment(i * dim, dim) = compute_to(spillover.middleCols(i * dim, dim));
57 from_spillover.segment(i * dim, dim) = compute_from(spillover.middleCols(i * dim, dim));
58 tot_spillover[i] = compute_tot(spillover.middleCols(i * dim, dim));
59 net_spillover.middleCols(i * dim, dim) = compute_net(spillover.middleCols(i * dim, dim));
60 }

409 void runGibbs(int window, int chain) {
410 for (int i = 0; i < num_burn; ++i) {
411 model[window][chain]->doWarmUp();
412 }
413 for (int i = num_burn; i < num_iter; ++i) {
414 model[window][chain]->doPosteriorDraws();
415 }
416 LdltRecords reg_record = model[window][chain]->returnLdltRecords(0, thin, sparse);
417 spillover[window][chain] = initialize_ctaspillover<LdltRecords>(lag, step, reg_record, -1, har_trans);
418 model[window][chain].reset();
419 }
420 void getSpillover(int window, int chain) {
421 spillover[window][chain]->computeSpillover();
422 to_sp[window][chain] = spillover[window][chain]->returnTo();
423 from_sp[window][chain] = spillover[window][chain]->returnFrom();
424 tot[window][chain] = spillover[window][chain]->returnTot();
425 net_sp[window][chain] = to_sp[window][chain] - from_sp[window][chain];
426 spillover[window][chain].reset();
427 }
428 void fit() {
429 if (num_chains == 1) {
430 #ifdef _OPENMP
431 #pragma omp parallel for num_threads(nthreads)
432 #endif
433 for (int window = 0; window < num_horizon; ++window) {
434 runGibbs(window, 0);
435 getSpillover(window, 0);
436 }
437 } else {
438 #ifdef _OPENMP
439 #pragma omp parallel for collapse(2) schedule(static, num_chains) num_threads(nthreads)
440 #endif
441 for (int window = 0; window < num_horizon; ++window) {
442 for (int chain = 0; chain < num_chains; ++chain) {
443 runGibbs(window, chain);
444 getSpillover(window, chain);
445 }
