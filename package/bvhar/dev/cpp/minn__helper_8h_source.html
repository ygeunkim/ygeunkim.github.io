9// Numerically Stable Log Marginal Likelihood Excluding Constant Term
10//
11// This function computes log of ML stable,
12// excluding the constant term.
13//
14// @param dim Dimension of the time series
15// @param num_design The number of the data matrix, \eqn{n = T - p}
16// @param prior_prec Prior precision of Matrix Normal distribution
17// @param prior_scale Prior scale of Inverse-Wishart distribution
18// @param mn_prec Posterior precision of Matrix Normal distribution
19// @param iw_scale Posterior scale of Inverse-Wishart distribution
20// @param posterior_shape Posterior shape of Inverse-Wishart distribution
21inline double compute_logml(int dim, int num_design, Eigen::MatrixXd prior_prec, Eigen::MatrixXd prior_scale,
22 Eigen::MatrixXd mn_prec, Eigen::MatrixXd iw_scale, int posterior_shape) {
23 Eigen::LLT<Eigen::MatrixXd> lltOfmn(prior_prec.inverse());
24 Eigen::MatrixXd chol_mn = lltOfmn.matrixL();
25 Eigen::MatrixXd stable_mat_a = chol_mn.transpose() * (mn_prec - prior_prec) * chol_mn;
26 Eigen::LLT<Eigen::MatrixXd> lltOfiw(prior_scale.inverse());
27 Eigen::MatrixXd chol_iw = lltOfiw.matrixL();
28 Eigen::MatrixXd stable_mat_b = chol_iw.transpose() * (iw_scale - prior_scale) * chol_iw;
29 // eigenvalues
30 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es_a(stable_mat_a);
31 Eigen::VectorXd a_eigen = es_a.eigenvalues();
32 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es_b(stable_mat_b);
33 Eigen::VectorXd b_eigen = es_b.eigenvalues();
34 // sum of log(1 + eigenvalues)
35 double a_term = a_eigen.array().log1p().sum();
36 double b_term = b_eigen.array().log1p().sum();
37 // result
38 return - num_design / 2.0 * log(prior_scale.determinant()) - dim / 2.0 * a_term - posterior_shape / 2.0 * b_term;
39}

41// Log of Joint Posterior Density of Hyperparameters
42//
43// This function computes the log of joint posterior density of hyperparameters.
44//
45// @param cand_gamma Candidate value of hyperparameters following Gamma distribution
46// @param cand_invgam Candidate value of hyperparameters following Inverse Gamma distribution
47// @param dim Dimension of the time series
48// @param num_design The number of the data matrix, \eqn{n = T - p}
49// @param prior_prec Prior precision of Matrix Normal distribution
50// @param prior_scale Prior scale of Inverse-Wishart distribution
51// @param mn_prec Posterior precision of Matrix Normal distribution
52// @param iw_scale Posterior scale of Inverse-Wishart distribution
53// @param posterior_shape Posterior shape of Inverse-Wishart distribution
54// @param gamma_shape Shape of hyperprior Gamma distribution
55// @param gamma_rate Rate of hyperprior Gamma distribution
56// @param invgam_shape Shape of hyperprior Inverse gamma distribution
57// @param invgam_scl Scale of hyperprior Inverse gamma distribution
58inline double jointdens_hyperparam(double cand_gamma, Eigen::VectorXd cand_invgam, int dim, int num_design,
59 Eigen::MatrixXd prior_prec, Eigen::MatrixXd prior_scale, int prior_shape,
60 Eigen::MatrixXd mn_prec, Eigen::MatrixXd iw_scale,
61 int posterior_shape, double gamma_shp, double gamma_rate, double invgam_shp, double invgam_scl) {
62 double res = compute_logml(dim, num_design, prior_prec, prior_scale, mn_prec, iw_scale, posterior_shape);
63 res += -dim * num_design / 2.0 * log(M_PI) +
64 lmgammafn((prior_shape + num_design) / 2.0, dim) -
65 lmgammafn(prior_shape / 2.0, dim); // constant term
66 res += gamma_dens(cand_gamma, gamma_shp, 1 / gamma_rate, true); // gamma distribution
67 for (int i = 0; i < cand_invgam.size(); i++) {
68 res += invgamma_dens(cand_invgam[i], invgam_shp, invgam_scl, true); // inverse gamma distribution
69 }
70 return res;
71}

73// Generating contemporaneous lambda of Minnesota-SV
74//
75// @param lambda lambda1 or lambda2
76// @param shape Gamma prior shape
77// @param rate Gamma prior rate
78// @param coef_vec Coefficients vector
79// @param coef_mean Prior mean
80// @param coef_prec Prior precision matrix
81// @param rng boost rng
82inline void minnesota_lambda(double& lambda, double& shape, double& rate, Eigen::Ref<Eigen::VectorXd> coef,
83 Eigen::Ref<Eigen::VectorXd> coef_mean, Eigen::Ref<Eigen::VectorXd> coef_prec,
84 BVHAR_BHRNG& rng) {
85 coef_prec.array() *= lambda;
86 // double gig_chi = (coef - coef_mean).squaredNorm();
87 double gig_chi = ((coef - coef_mean).array().square() * coef_prec.array()).sum();
88 lambda = sim_gig(shape - coef.size() / 2, 2 * rate, gig_chi, rng);
89 cut_param(lambda);
90 coef_prec.array() /= lambda;
91}

93inline double minnesota_logdens_scl(double& cand, Eigen::Ref<Eigen::VectorXd> coef,
94 Eigen::Ref<Eigen::VectorXd> coef_mean, Eigen::Ref<Eigen::VectorXd> coef_prec,
95 Eigen::VectorXi& grp_vec, std::set<int>& grp_id) {
96 double gaussian_kernel = 0;
97 int mn_size = 0;
98 for (int i = 0; i < coef.size(); ++i) {
99 if (grp_id.find(grp_vec[i]) != grp_id.end()) {
100 mn_size++;
101 gaussian_kernel += (coef[i] - coef_mean[i]) * (coef[i] - coef_mean[i]) * coef_prec[i];
102 }
103 }
104 return -(mn_size * log(cand) + gaussian_kernel / cand) / 2;
105}
