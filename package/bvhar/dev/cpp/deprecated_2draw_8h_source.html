10// Generating the Diagonal Component of Cholesky Factor in SSVS Gibbs Sampler
11//
12// In MCMC process of SSVS, this function generates the diagonal component \eqn{\Psi} from variance matrix
13//
14// @param sse_mat The result of \eqn{Z_0^T Z_0 = (Y_0 - X_0 \hat{A})^T (Y_0 - X_0 \hat{A})}
15// @param DRD Inverse of matrix product between \eqn{D_j} and correlation matrix \eqn{R_j}
16// @param shape Gamma shape parameters for precision matrix
17// @param rate Gamma rate parameters for precision matrix
18// @param num_design The number of sample used, \eqn{n = T - p}
19inline void ssvs_chol_diag(Eigen::VectorXd& chol_diag, Eigen::MatrixXd& sse_mat, Eigen::VectorXd& DRD,
20 Eigen::VectorXd& shape, Eigen::VectorXd& rate, int num_design, boost::mt19937& rng) {
21 int dim = sse_mat.cols();
22 int num_param = DRD.size();
23 Eigen::MatrixXd inv_DRD = Eigen::MatrixXd::Zero(num_param, num_param);
24 inv_DRD.diagonal() = 1 / DRD.array().square();
25 Eigen::VectorXd sse_colvec(dim - 1); // sj = (s1j, ..., s(j-1, j)) from SSE
26 shape.array() += (double)num_design / 2;
27 rate[0] += sse_mat(0, 0) / 2;
28 chol_diag[0] = sqrt(gamma_rand(shape[0], 1 / rate[0], rng)); // psi[11]^2 ~ Gamma(shape, rate)
29 int block_id = 0;
30 for (int j = 1; j < dim; j++) {
31 sse_colvec.segment(0, j) = sse_mat.block(0, j, j, 1); // (s1j, ..., sj-1,j)
32 rate[j] += (
33 sse_mat(j, j) -
34 sse_colvec.segment(0, j).transpose() *
35 (sse_mat.topLeftCorner(j, j) + inv_DRD.block(block_id, block_id, j, j)).llt().solve(Eigen::MatrixXd::Identity(j, j)) *
36 sse_colvec.segment(0, j)
37 ) / 2;
38 chol_diag[j] = sqrt(gamma_rand(shape[j], 1 / rate[j], rng)); // psi[jj]^2 ~ Gamma(shape, rate)
39 block_id += j;
40 }
41}

43// Generating the Off-Diagonal Component of Cholesky Factor in SSVS Gibbs Sampler
44//
45// In MCMC process of SSVS, this function generates the off-diagonal component \eqn{\Psi} of variance matrix
46//
47// @param sse_mat The result of \eqn{Z_0^T Z_0 = (Y_0 - X_0 \hat{A})^T (Y_0 - X_0 \hat{A})}
48// @param chol_diag Diagonal element of the cholesky factor
49// @param DRD Inverse of matrix product between \eqn{D_j} and correlation matrix \eqn{R_j}
50inline void ssvs_chol_off(Eigen::VectorXd& chol_off, Eigen::MatrixXd& sse_mat,
51 Eigen::VectorXd& chol_diag, Eigen::VectorXd& DRD, BHRNG& rng) {
52 int dim = sse_mat.cols();
53 int num_param = DRD.size();
54 Eigen::MatrixXd normal_variance(dim - 1, dim - 1);
55 Eigen::VectorXd sse_colvec(dim - 1); // sj = (s1j, ..., s(j-1, j)) from SSE
56 Eigen::VectorXd normal_mean(dim - 1);
57 // Eigen::VectorXd res(num_param);
58 Eigen::MatrixXd inv_DRD = Eigen::MatrixXd::Zero(num_param, num_param);
59 inv_DRD.diagonal() = 1 / DRD.array().square();
60 int block_id = 0;
61 for (int j = 1; j < dim; j++) {
62 sse_colvec.segment(0, j) = sse_mat.block(0, j, j, 1);
63 normal_variance.topLeftCorner(j, j) = (sse_mat.topLeftCorner(j, j) + inv_DRD.block(block_id, block_id, j, j)).llt().solve(Eigen::MatrixXd::Identity(j, j));
64 normal_mean.segment(0, j) = -chol_diag[j] * normal_variance.topLeftCorner(j, j) * sse_colvec.segment(0, j);
65 chol_off.segment(block_id, j) = vectorize_eigen(sim_mgaussian_chol(1, normal_mean.segment(0, j), normal_variance.topLeftCorner(j, j), rng));
66 block_id += j;
67 }

70// Filling Cholesky Factor Upper Triangular Matrix
71//
72// This function builds a cholesky factor matrix \eqn{\Psi} (upper triangular) using diagonal component vector and off-diagonal component vector.
73//
74// @param diag_vec Diagonal components
75// @param off_diagvec Off-diagonal components
76inline Eigen::MatrixXd build_chol(Eigen::VectorXd diag_vec, Eigen::VectorXd off_diagvec) {
77 int dim = diag_vec.size();
78 Eigen::MatrixXd res = Eigen::MatrixXd::Zero(dim, dim);
79 res.diagonal() = diag_vec; // psi
80 int id = 0; // length of eta = m(m-1)/2
81 // assign eta_j = (psi_1j, ..., psi_j-1,j)
82 for (int j = 1; j < dim; j++) {
83 for (int i = 0; i < j; i++) {
84 res(i, j) = off_diagvec[id + i]; // assign i-th row = psi_ij
85 }
86 id += j;
87 }
88 return res;
89}

169 coef = post_mean + llt_sig.matrixU().solve(res);

172// Generating the Coefficient Vector using Fast Sampling
173//
174// In MCMC process of Horseshoe prior, this function generates the coefficients vector.
175//
176// @param response_vec Response vector for vectorized formulation
177// @param design_mat Design matrix for vectorized formulation
178// @param shrink_mat Diagonal matrix made by global and local sparsity hyperparameters
179inline void horseshoe_fast_coef(Eigen::VectorXd& coef, Eigen::VectorXd response_vec, Eigen::MatrixXd design_mat,
180 Eigen::MatrixXd shrink_mat, BHRNG& rng) {
181 int num_coef = design_mat.cols(); // k^2 kp(+1)
182 int num_sur = response_vec.size(); // nk-dim
183 Eigen::MatrixXd sur_identity = Eigen::MatrixXd::Identity(num_sur, num_sur);
184 Eigen::VectorXd u_vec = vectorize_eigen(sim_mgaussian_chol(1, Eigen::VectorXd::Zero(num_coef), shrink_mat, rng));
185 Eigen::VectorXd delta_vec = vectorize_eigen(sim_mgaussian_chol(1, Eigen::VectorXd::Zero(num_sur), sur_identity, rng));
186 Eigen::VectorXd nu = design_mat * u_vec + delta_vec;
187 Eigen::VectorXd lin_solve = (design_mat * shrink_mat * design_mat.transpose() + sur_identity).llt().solve(
188 response_vec - nu
189 );
190 coef = u_vec + shrink_mat * design_mat.transpose() * lin_solve;
191}
