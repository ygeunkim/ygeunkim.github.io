670 BVHAR_STRING b_name = sparse ? "b_sparse_record" : "b_record";
671 initialize_record(reg_record, i, fit_record, include_mean, coef_name, a_name, c_name, b_name);
672 } else if (size_factor && !exogen) {
673 BVHAR_STRING lam_name = sparse ? "Lambda_sparse_record" : "Lambda_record";
674 initialize_record(reg_record, i, fit_record, include_mean, coef_name, a_name, c_name, lam_name);

686 BVHAR_STRING f_name = "F_record";

689 factor_updater = std::make_unique<CtaFactorNormalForecaster>(*dfm_record, step, response_mat.cols(), *size_factor);
690 } else {
691 // factor_updater = std::make_unique<CtaFactorNormalForecaster>(*dfm_record, step, response_mat.cols(), *size_factor);
692 }
693 } else {
694 if (*factor_lag == 0) {
695 factor_updater = std::make_unique<CtaFactorNormalForecaster>(step, response_mat.cols(), *size_factor);
696 } else {
697 // factor_updater = std::make_unique<CtaFactorNormalForecaster>(step, response_mat.cols(), *size_factor);
698 }
699 }
700 }
701 // std::unique_ptr<CtaExogenForecaster> exogen_updater;
702 // if (exogen) {
703 // exogen_updater = std::make_unique<CtaExogenForecaster>(*exogen_lag, *exogen, response_mat.cols());
704 // } else {
705 // exogen_updater = nullptr;
706 // }
707 if (har_trans && !activity) {
708 forecaster_ptr[i] = std::make_unique<CtaVharForecaster<BaseForecaster>>(
709 *reg_record, step, response_mat,
710 *har_trans, ord,
711 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
712 sv, std::move(exogen_updater), std::move(factor_updater)
713 );
714 } else if (!har_trans && !activity) {
715 forecaster_ptr[i] = std::make_unique<CtaVarForecaster<BaseForecaster>>(
716 *reg_record, step, response_mat,
717 ord,
718 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
719 sv, std::move(exogen_updater), std::move(factor_updater)
720 );
721 } else if (har_trans && activity) {
722 forecaster_ptr[i] = std::make_unique<CtaVharSelectForecaster<BaseForecaster>>(
723 *reg_record, level, step, response_mat,
724 *har_trans, ord,
725 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
726 sv, std::move(exogen_updater), std::move(factor_updater)
727 );
728 } else {
729 forecaster_ptr[i] = std::make_unique<CtaVarSelectForecaster<BaseForecaster>>(
730 *reg_record, level, step, response_mat,
731 ord,
732 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
733 sv, std::move(exogen_updater), std::move(factor_updater)
734 );
735 }
736 }
