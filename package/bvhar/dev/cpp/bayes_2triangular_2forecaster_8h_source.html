660 BVHAR_STRING a_name = sparse ? "a_sparse_record" : "a_record";
661 BVHAR_STRING c_name = sparse ? "c_sparse_record" : "c_record";
662 for (int i = 0; i < num_chains; ++i) {
663 std::unique_ptr<Records> reg_record;
664 if (exogen && size_factor) {
665 BVHAR_STRING b_name = sparse ? "b_sparse_record" : "b_record";
666 BVHAR_STRING lam_name = sparse ? "Lambda_sparse_record" : "Lambda_record";
667 initialize_record(reg_record, i, fit_record, include_mean, coef_name, a_name, c_name, b_name, lam_name);
668 } else if (exogen && !size_factor) {
669 BVHAR_STRING b_name = sparse ? "b_sparse_record" : "b_record";
670 initialize_record(reg_record, i, fit_record, include_mean, coef_name, a_name, c_name, b_name);
671 } else if (size_factor && !exogen) {
672 BVHAR_STRING lam_name = sparse ? "Lambda_sparse_record" : "Lambda_record";
673 initialize_record(reg_record, i, fit_record, include_mean, coef_name, a_name, c_name, lam_name);
674 } else {
675 initialize_record(reg_record, i, fit_record, include_mean, coef_name, a_name, c_name);
676 }
677 BVHAR_OPTIONAL<std::unique_ptr<CtaExogenForecaster>> exogen_updater = BVHAR_NULLOPT;
678 BVHAR_OPTIONAL<std::unique_ptr<CtaFactorForecaster>> factor_updater = BVHAR_NULLOPT;
679 if (exogen) {
680 exogen_updater = std::make_unique<CtaExogenForecaster>(*exogen_lag, *exogen, response_mat.cols());
681 }
682 if (size_factor) {
683 std::unique_ptr<DfmRecords> dfm_record;
684 if (*factor_insample) {
685 BVHAR_STRING f_name = "F_record";
686 initialize_dfmrecord(dfm_record, i, fit_record, f_name);
687 if (*factor_lag == 0) {
688 factor_updater = std::make_unique<CtaFactorNormalForecaster>(*dfm_record, step, response_mat.cols(), *size_factor);
689 } else {
690 // factor_updater = std::make_unique<CtaFactorNormalForecaster>(*dfm_record, step, response_mat.cols(), *size_factor);
691 }
692 } else {
693 if (*factor_lag == 0) {
694 factor_updater = std::make_unique<CtaFactorNormalForecaster>(step, response_mat.cols(), *size_factor);
695 } else {
696 // factor_updater = std::make_unique<CtaFactorNormalForecaster>(step, response_mat.cols(), *size_factor);
697 }
698 }
699 }
700 // std::unique_ptr<CtaExogenForecaster> exogen_updater;
701 // if (exogen) {
702 // exogen_updater = std::make_unique<CtaExogenForecaster>(*exogen_lag, *exogen, response_mat.cols());
703 // } else {
704 // exogen_updater = nullptr;
705 // }
706 if (har_trans && !activity) {
707 forecaster_ptr[i] = std::make_unique<CtaVharForecaster<BaseForecaster>>(
708 *reg_record, step, response_mat,
709 *har_trans, ord,
710 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
711 sv, std::move(exogen_updater), std::move(factor_updater)
712 );
713 } else if (!har_trans && !activity) {
714 forecaster_ptr[i] = std::make_unique<CtaVarForecaster<BaseForecaster>>(
715 *reg_record, step, response_mat,
716 ord,
717 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
718 sv, std::move(exogen_updater), std::move(factor_updater)
719 );
720 } else if (har_trans && activity) {
721 forecaster_ptr[i] = std::make_unique<CtaVharSelectForecaster<BaseForecaster>>(
722 *reg_record, level, step, response_mat,
723 *har_trans, ord,
724 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
725 sv, std::move(exogen_updater), std::move(factor_updater)
726 );
727 } else {
728 forecaster_ptr[i] = std::make_unique<CtaVarSelectForecaster<BaseForecaster>>(
729 *reg_record, level, step, response_mat,
730 ord,
731 include_mean, stable, static_cast<unsigned int>(seed_chain[i]),
732 sv, std::move(exogen_updater), std::move(factor_updater)
733 );
734 }
735 }
