105 RealType operator()(URNG& urng) const {
106#ifndef BOOST_NO_STDC_NAMESPACE
107 using std::sqrt;
108#endif
109 RealType w = _alpha * chi_squared_distribution<RealType>(result_type(1))(urng);
110 RealType cand = _alpha + _c * (w - sqrt(w * (result_type(4) * _beta + w)));

114 }
115 return _alpha * _alpha / cand;

259 template<class URNG>
260 RealType operator()(URNG& urng) const {
261#ifndef BOOST_NO_STDC_NAMESPACE
262 using std::sqrt;
263 using std::log;
264 using std::min;
265 using std::exp;
266 using std::cosh;
267#endif
268 RealType t = result_type(1);
269 RealType s = result_type(1);
270 RealType log_concave = -psi(result_type(1));
271 if (log_concave >= result_type(.5) && log_concave <= result_type(2)) {
272 t = result_type(1);
273 } else if (log_concave > result_type(2)) {
274 t = sqrt(result_type(2) / (_alpha + _abs_p));
275 } else if (log_concave < result_type(.5)) {
276 t = log(result_type(4) / (_alpha + result_type(2) * _abs_p));
277 }
278 log_concave = -psi(result_type(-1));
279 if (log_concave >= result_type(.5) && log_concave <= result_type(2)) {
280 s = result_type(1);
281 } else if (log_concave > result_type(2)) {
282 s = sqrt(result_type(4) / (_alpha * cosh(1) + _abs_p));
283 } else if (log_concave < result_type(.5)) {
284 s = min(result_type(1) / _abs_p, log(result_type(1) + result_type(1) / _alpha + sqrt(result_type(1) / (_alpha * _alpha) + result_type(2) / _alpha)));
285 }
286 RealType eta = -psi(t);
287 RealType zeta = -psi_deriv(t);
288 RealType theta = -psi(-s);
289 RealType xi = psi_deriv(-s);
290 RealType p = result_type(1) / xi;
291 RealType r = result_type(1) / zeta;
292 RealType t_deriv = t - r * eta;
293 RealType s_deriv = s - p * theta;
294 RealType q = t_deriv + s_deriv;
295 RealType u = result_type(0);
296 RealType v = result_type(0);
297 RealType w = result_type(0);
298 RealType cand = result_type(0);
299 do {
300 u = uniform_01<RealType>()(urng);
301 v = uniform_01<RealType>()(urng);
302 w = uniform_01<RealType>()(urng);
303 if (u < q / (p + q + r)) {
304 cand = -s_deriv + q * v;
305 } else if (u < (q + r) / (p + q + r)) {
306 cand = t_deriv - r * log(v);
307 } else {
308 cand = -s_deriv + p * log(v);
309 }
310 } while (w * chi(cand, s, t, s_deriv, t_deriv, eta, zeta, theta, xi) > exp(psi(cand)));
311 cand = (_abs_p / _omega + sqrt(result_type(1) + _abs_p * _abs_p / (_omega * _omega))) * exp(cand);
312 return _p > 0 ? cand * sqrt(_b / _a) : sqrt(_b / _a) / cand;
313 }

362 void init() {
363#ifndef BOOST_NO_STDC_NAMESPACE
364 using std::abs;
365 using std::sqrt;
366#endif
367 _abs_p = abs(_p);
368 _omega = sqrt(_a * _b); // two-parameter representation (p, omega)
369 _alpha = sqrt(_omega * _omega + _abs_p * _abs_p) - _abs_p;
370 }
371
372 result_type psi(const RealType& x) const {
373#ifndef BOOST_NO_STDC_NAMESPACE
374 using std::cosh;
375 using std::exp;
376#endif
377 return -_alpha * (cosh(x) - result_type(1)) - _abs_p * (exp(x) - x - result_type(1));
378 }
379
380 result_type psi_deriv(const RealType& x) const {
381#ifndef BOOST_NO_STDC_NAMESPACE
382 using std::sinh;
383 using std::exp;
384#endif
385 return -_alpha * sinh(x) - _abs_p * (exp(x) - result_type(1));
386 }
387
388 static result_type chi(const RealType& x,
389 const RealType& s, const RealType& t,
390 const RealType& s_deriv, const RealType& t_deriv,
391 const RealType& eta, const RealType& zeta, const RealType& theta, const RealType& xi) {
392#ifndef BOOST_NO_STDC_NAMESPACE
393 using std::exp;
394#endif
395 if (x >= -s_deriv && x <= t_deriv) {
396 return result_type(1);
397 } else if (x > t_deriv) {
398 return exp(-eta - zeta * (x - t));
399 }
400 return exp(-theta + xi * (x + s));
401 }

464inline double mgammafn(double x, int p) {
465 if (p < 1) {
466 BVHAR_STOP("'p' should be larger than or same as 1.");
467 }
468 if (x <= 0) {
469 BVHAR_STOP("'x' should be larger than 0.");
470 }
471 if (p == 1) {
472 return gammafn(x);
473 }
474 if (2 * x < p) {
475 BVHAR_STOP("'x / 2' should be larger than 'p'.");
476 }
477 double res = pow(M_PI, p * (p - 1) / 4.0);
478 for (int i = 0; i < p; i++) {
479 res *= gammafn(x - i / 2.0); // x + (1 - j) / 2
480 }
481 return res;
482}

519// Density of Inverse Gamma Distribution
520//
521// Compute the pdf of Inverse Gamma distribution
522//
523// @param x non-negative argument
524// @param shp Shape of the distribution
525// @param scl Scale of the distribution
526// @param lg If true, return log(f)
527inline double invgamma_dens(double x, double shp, double scl, bool lg) {
528 if (x < 0 ) {
529 BVHAR_STOP("'x' should be larger than 0.");
530 }
531 if (shp <= 0 ) {
532 BVHAR_STOP("'shp' should be larger than 0.");
533 }
534 if (scl <= 0 ) {
535 BVHAR_STOP("'scl' should be larger than 0.");
536 }
537 double res = pow(scl, shp) * pow(x, -shp - 1) * exp(-scl / x) / bvhar::gammafn(shp);
538 if (lg) {
539 return log(res);
540 }
541 return res;
542}
