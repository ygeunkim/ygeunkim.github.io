102 RealType operator()(URNG& urng) const {
103#ifndef BOOST_NO_STDC_NAMESPACE
104 using std::sqrt;
105#endif
106 RealType w = _alpha * chi_squared_distribution<RealType>(result_type(1))(urng);
107 RealType cand = _alpha + _c * (w - sqrt(w * (result_type(4) * _beta + w)));
108 RealType u = uniform_01<RealType>()(urng);
109 if (u < _alpha / (_alpha + cand)) {
110 return cand;
111 }
112 return _alpha * _alpha / cand;
113 }

236 template<class URNG>
237 RealType operator()(URNG& urng) const {
238#ifndef BOOST_NO_STDC_NAMESPACE
239 using std::sqrt;
240 using std::log;
241 using std::min;
242 using std::exp;
243 using std::cosh;
244#endif
245 RealType t = result_type(1);
246 RealType s = result_type(1);
247 RealType log_concave = -psi(result_type(1));
248 if (log_concave >= result_type(.5) && log_concave <= result_type(2)) {
249 t = result_type(1);
250 } else if (log_concave > result_type(2)) {
251 t = sqrt(result_type(2) / (_alpha + _abs_p));
252 } else if (log_concave < result_type(.5)) {
253 t = log(result_type(4) / (_alpha + result_type(2) * _abs_p));
254 }
255 log_concave = -psi(result_type(-1));
256 if (log_concave >= result_type(.5) && log_concave <= result_type(2)) {
257 s = result_type(1);
258 } else if (log_concave > result_type(2)) {
259 s = sqrt(result_type(4) / (_alpha * cosh(1) + _abs_p));
260 } else if (log_concave < result_type(.5)) {
261 s = min(result_type(1) / _abs_p, log(result_type(1) + result_type(1) / _alpha + sqrt(result_type(1) / (_alpha * _alpha) + result_type(2) / _alpha)));
262 }
263 RealType eta = -psi(t);
264 RealType zeta = -psi_deriv(t);
265 RealType theta = -psi(-s);
266 RealType xi = psi_deriv(-s);
267 RealType p = result_type(1) / xi;
268 RealType r = result_type(1) / zeta;
269 RealType t_deriv = t - r * eta;
270 RealType s_deriv = s - p * theta;
271 RealType q = t_deriv + s_deriv;
272 RealType u = result_type(0);
273 RealType v = result_type(0);
274 RealType w = result_type(0);
275 RealType cand = result_type(0);
276 do {
277 u = uniform_01<RealType>()(urng);
278 v = uniform_01<RealType>()(urng);
279 w = uniform_01<RealType>()(urng);
280 if (u < q / (p + q + r)) {
281 cand = -s_deriv + q * v;
282 } else if (u < (q + r) / (p + q + r)) {
283 cand = t_deriv - r * log(v);
284 } else {
285 cand = -s_deriv + p * log(v);
286 }
287 } while (w * chi(cand, s, t, s_deriv, t_deriv, eta, zeta, theta, xi) > exp(psi(cand)));
288 cand = (_abs_p / _omega + sqrt(result_type(1) + _abs_p * _abs_p / (_omega * _omega))) * exp(cand);
289 return _p > 0 ? cand * sqrt(_b / _a) : sqrt(_b / _a) / cand;
290 }

339 void init() {
340#ifndef BOOST_NO_STDC_NAMESPACE
341 using std::abs;
342 using std::sqrt;
343#endif
344 _abs_p = abs(_p);
345 _omega = sqrt(_a * _b); // two-parameter representation (p, omega)
346 _alpha = sqrt(_omega * _omega + _abs_p * _abs_p) - _abs_p;
347 }
348
349 result_type psi(const RealType& x) const {
350#ifndef BOOST_NO_STDC_NAMESPACE
351 using std::cosh;
352 using std::exp;
353#endif
354 return -_alpha * (cosh(x) - result_type(1)) - _abs_p * (exp(x) - x - result_type(1));
355 }
356
357 result_type psi_deriv(const RealType& x) const {
358#ifndef BOOST_NO_STDC_NAMESPACE
359 using std::sinh;
360 using std::exp;
361#endif
362 return -_alpha * sinh(x) - _abs_p * (exp(x) - result_type(1));
363 }

365 static result_type chi(const RealType& x,
366 const RealType& s, const RealType& t,
367 const RealType& s_deriv, const RealType& t_deriv,
368 const RealType& eta, const RealType& zeta, const RealType& theta, const RealType& xi) {
369#ifndef BOOST_NO_STDC_NAMESPACE
370 using std::exp;
371#endif
372 if (x >= -s_deriv && x <= t_deriv) {
373 return result_type(1);
374 } else if (x > t_deriv) {
375 return exp(-eta - zeta * (x - t));
376 }
377 return exp(-theta + xi * (x + s));
378 }

446inline double mgammafn(double x, int p) {
447 if (p < 1) {
448 STOP("'p' should be larger than or same as 1.");
449 }
450 if (x <= 0) {
451 STOP("'x' should be larger than 0.");
452 }
453 if (p == 1) {
454 return gammafn(x);
455 }
456 if (2 * x < p) {
457 STOP("'x / 2' should be larger than 'p'.");
458 }
459 double res = pow(M_PI, p * (p - 1) / 4.0);
460 for (int i = 0; i < p; i++) {
461 res *= gammafn(x - i / 2.0); // x + (1 - j) / 2
462 }
463 return res;
464}

476//
477// Compute log of multivariate gamma function numerically
478//
479// @param x Double, non-negative argument
480// @param p Integer, dimension
481inline double lmgammafn(double x, int p) {
482 // if (p < 1) {
483 // Rcpp::stop("'p' should be larger than or same as 1.");
484 // }
485 // if (x <= 0) {
486 // Rcpp::stop("'x' should be larger than 0.");
487 // }

492 // Rcpp::stop("'x / 2' should be larger than 'p'.");
493 // }
494 double res = p * (p - 1) / 4.0 * log(M_PI);
495 for (int i = 0; i < p; i++) {
496 res += lgammafn(x - i / 2.0);

501// Density of Inverse Gamma Distribution
502//
503// Compute the pdf of Inverse Gamma distribution
504//
505// @param x non-negative argument
506// @param shp Shape of the distribution
507// @param scl Scale of the distribution
508// @param lg If true, return log(f)
509inline double invgamma_dens(double x, double shp, double scl, bool lg) {
510 if (x < 0 ) {
511 STOP("'x' should be larger than 0.");
512 }
513 if (shp <= 0 ) {
514 STOP("'shp' should be larger than 0.");
515 }
516 if (scl <= 0 ) {
517 STOP("'scl' should be larger than 0.");
518 }
519 double res = pow(scl, shp) * pow(x, -shp - 1) * exp(-scl / x) / bvhar::gammafn(shp);
520 if (lg) {
521 return log(res);
522 }
523 return res;
524}
